\begin{aktivitas}
  \item Rancang basis pengetahuan untuk hierarki organisasi kampus (rektor, dekan, ketua prodi, dosen, mahasiswa). Daftarkan minimal 8 fakta dan 4 aturan.
  \item Demonstrasikan penalaran forward chaining untuk menjawab query ``Siapa atasan langsung dari mahasiswa X?'' menggunakan KB yang Anda buat.
  \item Modelkan aturan perpustakaan: ``Mahasiswa yang terdaftar dan tidak memiliki denda dapat meminjam buku.'' Tuliskan dalam notasi FOL dengan fakta dan aturan.
  \item Tunjukkan penerapan keempat aturan inferensi kuantor (UI, UG, EI, EG) pada contoh yang Anda buat.
  \item Bandingkan penerapan CWA dan OWA pada basis pengetahuan perpustakaan: apa yang terjadi jika $Terdaftar(MahasiswaY)$ tidak ada dalam KB?
  \item Identifikasi minimal 2 batasan FOL yang relevan dalam pemodelan sistem informasi akademik. Jelaskan bagaimana batasan ini dapat diatasi.
\end{aktivitas}

\begin{latihan}
  \item Tuliskan translasi fakta berikut: ``Ali suka matematika. Budi suka informatika. Cici suka fisika. Ali suka informatika.'' dalam format $Suka(x, y)$.
  \item Bangun aturan $Rekan(x, y)$: ``$x$ adalah rekan $y$ jika keduanya menyukai mata kuliah yang sama dan merupakan entitas berbeda.'' Tunjukkan proses inferensi untuk domain $\{Ali, Budi, Cici\}$.
  \item Diberikan aturan $\forall x, y\, (OrangTua(x, y) \wedge Laki(x) \rightarrow Ayah(x, y))$. Jika KB berisi $OrangTua(Hasan, Ali)$ dan $Laki(Hasan)$, tunjukkan langkah-langkah inferensi untuk membuktikan $Ayah(Hasan, Ali)$.
  \item Modelkan aturan e-commerce: ``Pengguna yang login, terverifikasi, dan memiliki saldo $\geq$ harga item dapat melakukan pembelian.'' Tuliskan dalam FOL dan demonstrasikan inferensi.
  \item Jelaskan perbedaan antara Existential Instantiation dan Universal Instantiation. Berikan contoh penggunaan masing-masing.
  \item Apa yang terjadi jika asumsi konklusi dalam backward chaining berujung pada fakta yang tidak ada dalam KB di bawah CWA?
  \item \textbf{Refleksi}: Berikan contoh masalah kompleks di dunia nyata yang lebih mudah direpresentasikan menggunakan basis pengetahuan (pendekatan deklaratif) dibandingkan algoritma imperatif.
  \item \textbf{Refleksi}: Mengapa basis data relasional (SQL) mengadopsi CWA, sementara web semantik (OWL) mengadopsi OWA? Apa implikasinya bagi desain sistem?
\end{latihan}

\begin{asesmen}
\textbf{Instrumen Evaluasi Kinerja untuk Sub-CPMK 2.3--2.5}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Dalam skema basis pengetahuan, elemen individual statis yang menyatakan kebenaran spesifik disebut:
  \begin{enumerate}
    \item Aturan
    \item Fakta atomik (ground atom)
    \item Variabel terikat
    \item Kuantor eksistensial
  \end{enumerate}
  Jawaban: (b)

  \item Untuk memodelkan aturan umum ``Semua manusia adalah mortal'' dalam FOL, instrumen yang tepat adalah:
  \begin{enumerate}
    \item Konjungsi murni tanpa parameter
    \item Implikasi berkuantor universal ($\forall x\, (P(x) \rightarrow Q(x))$)
    \item Operasi biimplikasi
    \item Hukum De Morgan
  \end{enumerate}
  Jawaban: (b)

  \item Universal Instantiation memungkinkan kita:
  \begin{enumerate}
    \item Menggeneralisasi fakta spesifik menjadi aturan universal
    \item Menurunkan fakta spesifik $P(c)$ dari aturan universal $\forall x\, P(x)$
    \item Membuktikan keberadaan entitas dengan properti tertentu
    \item Menggabungkan dua aturan menjadi satu
  \end{enumerate}
  Jawaban: (b)

  \item Di bawah Asumsi Dunia Tertutup (CWA), jika $Laki(Budi)$ tidak ada dalam KB, maka:
  \begin{enumerate}
    \item $Laki(Budi)$ bernilai benar
    \item $Laki(Budi)$ bernilai tidak diketahui
    \item $Laki(Budi)$ dianggap salah
    \item Sistem menghasilkan error
  \end{enumerate}
  Jawaban: (c)

  \item Strategi penalaran yang memulai dari query dan mencari aturan yang relevan secara mundur disebut:
  \begin{enumerate}
    \item Forward chaining
    \item Backward chaining
    \item Modus ponens
    \item Universal generalization
  \end{enumerate}
  Jawaban: (b)
\end{enumerate}

\textbf{B. Tugas Praktik}
\begin{enumerate}
  \item Buat skenario basis pengetahuan untuk organisasi kepanitiaan mahasiswa. Daftarkan $\Delta$ (fakta) dan $\Gamma$ (aturan). Tunjukkan alur inferensi untuk mendeduksi ``Siapa yang bertanggung jawab atas tugas X?''
  \item Bandingkan hasil query pada KB yang sama di bawah CWA dan OWA. Berikan contoh query yang menghasilkan jawaban berbeda.
\end{enumerate}

\textbf{Rubrik Penilaian}: Lihat Lampiran.
\end{asesmen}

\begin{rangkuman}
Bab ini membahas penerapan logika predikat orde pertama untuk memodelkan masalah komputasi dalam bentuk basis pengetahuan.

\textbf{Poin Kunci:}
\begin{itemize}
  \item \textbf{Basis Pengetahuan} terdiri dari fakta ($\Delta$, pernyataan atomik tanpa variabel) dan aturan ($\Gamma$, implikasi berkuantor universal). Pemodelan memerlukan identifikasi entitas, properti, dan relasi dari domain.
  \item \textbf{Arity} predikat menentukan jumlah argumennya: arity-1 untuk properti, arity-2 untuk relasi biner, dan arity-$n$ untuk relasi multi-argumen.
  \item \textbf{Empat aturan inferensi kuantor} (UI, UG, EI, EG) memungkinkan penalaran formal dengan membuka atau memasang kuantor.
  \item \textbf{Resolusi query} menggunakan forward chaining (data-driven) atau backward chaining (goal-driven) untuk menurunkan kesimpulan baru dari fakta dan aturan.
  \item \textbf{Asumsi Dunia Tertutup} (CWA) menganggap fakta yang tidak ada dalam KB sebagai salah, sementara \textbf{Asumsi Dunia Terbuka} (OWA) menganggapnya tidak diketahui.
  \item FOL memiliki batasan dalam menangani ketidakpastian, perubahan pengetahuan, dan kompleksitas komputasi, tetapi tetap menjadi fondasi bagi representasi pengetahuan modern.
\end{itemize}

\textbf{Kata Kunci}: basis pengetahuan, fakta, aturan, predikat, arity, instansiasi universal, jeneralisasi, forward chaining, backward chaining, CWA, OWA
\end{rangkuman}
