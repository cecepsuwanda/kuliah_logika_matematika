\begin{aktivitas}
  \item Diskusikan dalam kelompok: Amati acara perdebatan politik di televisi masa kini. Temukan minimal dua kalimat retoris debat yang sekilas terdengar memukau padahal aslinya murni cacat nalar jenis \textit{Fallacy of Affirming the Consequent}. Tuliskan transkripnya beserta bedah kelemahannya.
  \item Tulis kembali silogisme \textit{"Jika hujan maka jalanan licin. Jika jalanan licin kendaraan merayap"} ke dalam kode linear aljabar proposisi. Buktikan argumen \textbf{Silogisme Hipotetis} transisinya itu sukses melalui tes kesucian Tabel Kebenaran Tautologi 8-Baris!
  \item Berikan contoh sehari-hari yang valid di mana Jurus \textit{Penambahan Disjungtif} (Addition) digunakan oleh seseorang untuk memperkuat gertakan (meskipun tampak tidak selaras).
\end{aktivitas}

\begin{latihan}
  \item Tentukan jenis Aturan Inferensi yang bersembunyi di balik argumen ringkas berikut:
  \begin{enumerate}
    \item L: "Hanya jika engkau kaya, engkau bebas merdeka berkelana." \\
          M: "Nyatanya hari ini engkau dipenjara." \\
          K: "Maka engkau pasti miskin papa tertindas."
    \item L: "Ikan paus terdampar lapar atau ia menari gembira." \\
          M: "Nelayan membuktikan ia perutnya kenyang tak lapar." \\
          K: "Pasti paus nan agung itu sedang menari gembira."
  \end{enumerate}
  \item Tunjukkan skema alur barisan Pembuktian Formal Berantai (\textit{Formal Proof of Validity}) memakai format tabel, dari himpunan tiga Premis Majemuk berikut menuju TUDUHAN AKHIR KESIMPULAN KONKLUSI $(\neg P)$. Bebas gunakan Modus Ponens, Tollens dsb.
  \begin{itemize}
      \item Premis 1 Acuan: $(P \lor Q) \rightarrow R$
      \item Premis 2 Fakta: $S \rightarrow \neg R$
      \item Premis 3 Temuan Nyata: $S$ \;\;\textbf{(TUDUHAN AKHIR DICARI: $\therefore \neg P$)}
  \end{itemize}
  \item \textbf{Refleksi}: Mengapa Modus Ponens $(p \rightarrow q), p \vdash q$ terasa begitu natural bak nafas detak jantung penalaran manusia sehari-hari? Berikan alasan kognitifnya! Mengapa sebaliknya, \textit{Fallacy of Denying the Antecedent} kerap menjerat kaum awam?
\end{latihan}

\begin{asesmen}
\textbf{Instrumen Evaluasi Kinerja untuk Sub-CPMK 1.3}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Struktur argumen $(p \lor q)$ dipadukan $\neg p$ melahirkan konklusi sah pasti $q$, di pentas hukum matematika dinamakan jurus:
  \begin{enumerate}
    \item Silogisme Disjungtif
    \item Silogisme Hipotetis
    \item Resolusi AI Terapan
    \item Simplifikasi Konjungsi
  \end{enumerate}
  Jawaban: (a)

  \item Menimbang kepastian formula $(P \rightarrow Q)$, manakala seorang politisi nekat bersandar pada realita suksesnya si Akibat $Q$, lalu lantang mendeklarasikan bahwa $P$ otomatis sah! Hal keliru itu dijuluki:
  \begin{enumerate}
    \item Fallacy of Denying Antecedent
    \item Tautologi Negasi Berantai
    \item Fallacy of Affirming the Consequent
    \item Bukti Langsung 
  \end{enumerate}
  Jawaban: (c)

  \item Aturan Simplifikasi hanya diizinkan mencaplok memakan korban klausa yang saling terikat oleh lilitan tali tambang operator:
  \begin{enumerate}
    \item Implikasi mutlak ($\rightarrow$)
    \item Konjungsi ($\land$)
    \item Disjungsi ($\lor$)
    \item Ekivalen ($\leftrightarrow$)
  \end{enumerate}
  Jawaban: (b)

  \item Mesin Pemrosesan PROLOG (Kecerdasan Buatan) amat purba menggantungkan napas pencarian basis datanya di pundak peleburan konflik klausa $OR$ yang disingkat:
  \begin{enumerate}
    \item Algoritma Silogisme Modus Ponens
    \item Penyederhanaan Tollens
    \item Resolvent / Aturan Resolusi
    \item Fallacy of Composition
  \end{enumerate}
  Jawaban: (c)
\end{enumerate}

\textbf{B. Tugas Uji Komprehensif}
\begin{itemize}
  \item Tersaji Premis: "Kalau saya tak melunasi pajak, mobil saya diderek DLLAJ". Fakta sore ini mobil Anda lengang aman bersandar di garasi tanpa derek setitik pun. Rancang formulasi model premisnya, lalu hantam memakai salah satu Modus andalan hingga membongkar rahasia status Pajak Anda di mata publik! Sebutkan nama aturan spesifiknya.
\end{itemize}

\end{asesmen}

\begin{rangkuman}
Bab 5 menobatkan puncak utilitas terapan dari ilmu aljabar Proposisi: \textbf{Menarik Kesimpulan Inferensi Sah tanpa cacat Fallacy}.

\textbf{Poin Kunci:}
\begin{itemize}
  \item \textbf{Argumen Valid} menampakkan wajah komposit matematis bilamana rakitan selang lurusnya \textit{(Premis 1 $\land$ Premis 2 $\ldots$) $\rightarrow$ Konklusi} sejati dihukum lolos sensor \textbf{Tautologi}.
  \item \textbf{Modus Ponens} membina penetapan akibat (pujian), sedangkan saudarinya \textbf{Modus Tollens} menebas gugur akar sebab imbas gagalnya suatu impian akibat berdasar sifat \textit{Kontraposisi}.
  \item \textbf{Keluarga Silogisme} diwakili gerbong estafet energi murni (Hipotetis) atau pemaksaan pilihan pahit (Disjungtif).
  \item \textbf{Operasi Resolusi} menahbiskan dirinya tonggak utama pelacakan *Knowledge Base AI Programming* (Sistem Pakar).
  \item \textbf{Logical Fallacies Formal} menebar teror via persembunyian dua penipuan akbar kembar: Menetapkan Si Akibat (Menyontek Ponens) serta Menyangkal Sang Penyebab (Menyontek Tollens).
  \item Pengerjaan \textbf{Deduksi Formal} mengajarkan mahasiwa tata krama justifikasi teliti setiap lembar baris baris algoritma pergerakannya demi melacak ujung sejati Konklusi akhir.
\end{itemize}

\textbf{Kata Kunci}: Inferensi, Modus Ponens, Modus Tollens, Silogisme Hipotetis/Disjungtif, Aturan Resolusi, Kesesatan Berpikir (\textit{Fallacies}), Bukti Berantai (\textit{Formal Proof of Validity}), Tautologi Argumen.
\end{rangkuman}
