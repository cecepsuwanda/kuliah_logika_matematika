\section{Panduan Praktis Translasi Bahasa Alami}

Translasi statemen bahasa keseharian manusia, ke alam rumus Logika Predikat acap kali dinodai kesalahpahaman karena ambiguitas semantik kalimat verbal. Bagian ini merupakan matriks pedoman (ringkasan instan) kompas penerjemahan berbasis korelasi kuantor dan operator logika, guna menangkal kesalahan gramatika pemodelan fundamental \cite{rosen2019}.

\subsection{Kamus Konstruksi Proposisi}

\subsection{Contoh Translasi Kasus Jamak}

\begin{contoh}
Diberikan domain $\mathcal{U} = \text{semua makhluk}$. Predikat dasar: \\
$A(x)$ = $x$ adalah atlet \\
$S(x)$ = $x$ berbadan sehat \\
$K(x)$ = $x$ perokok

Lakukan translasi formal:
\begin{enumerate}
    \item \textbf{Setiap atlet pasti sehat.} \\
    $\forall x \, (A(x) \rightarrow S(x))$
    \item \textbf{Seorang perokok takkan pernah menjadi atlet.} \\
    "Takkan pernah" bermakna Tak Satupun. Rujuk nomor 3 di tabel. \\
    $\forall x \, (K(x) \rightarrow \neg A(x))$
    \item \textbf{Beberapa atlet sehat, yang bodohnya juga perokok.} \\
    "Beberapa" + konjungsi bertumpuk. \\
    $\exists x \, (A(x) \land S(x) \land K(x))$
    \item \textbf{Sebagian atlet ada yang sakit-sakitan (tidak sehat).} \\
    $\exists x \, (A(x) \land \neg S(x))$
    \item \textbf{Syarat telak sehat secara utuh (di organisasi ini), adalah ia tak merokok.} \\
    Hanya yang tak merokok bersertifikat sehat. Jadi: "setiap yg direkomendasikan sehat $\implies$ harus bebas tembakau". \\
    $\forall x \, (S(x) \rightarrow \neg K(x))$
\end{enumerate}
\end{contoh}
