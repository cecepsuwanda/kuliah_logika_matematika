\begin{latihan}
  \item Buatlah Tabel kebenaran kolosal 8-baris (untuk elemen $p, q, r$) membedah kepalsuan atau kesahihan Hukum Distributif bersilang gila berikut: $p \vee (q \wedge r) \equiv (p \vee q) \wedge (p \vee r)$. Yakinkan hakim matematika bahwa kolom final mereka berdua sungguh mutlak terlahir kembar siam!
  \item Gunakan pisau bedah Aljabar Proposisi murni (Hukum De Morgan, Asosiatif, Negasi Ganda dll) tanpa coretan Tabel setitik pun untuk meringkas rumus membengkak ini perlahan bertahap turun:
  $\neg( (\neg p \wedge q) \vee (p \vee \neg q) )$
  \item Sebuah barisan *Software Code* warisan *backend developer* lama memilah transaksi *e-commerce* seperti ini: 
  `if (saldo\_cukup and (bukan kredit or bukan saldo\_cukup))`. 
  Translasikan teks kusut itu ke variabel Logika $s, k$, lalu sabet memusnahkan keluwesannya! Tuliskan sisa \textit{Refactoring}-nya yang bersih.
  \item \textbf{Refleksi Singkat:} Di manakah ranah kekuatan dominan pengerjaan model bedah manipulasi Aljabar bila disandingkan bertarung dengan reliabilitas mutlak ala Tabel Kebenaran *Brute-Force*? Kapan Anda seyogyanya memilih tabel, dan kapan Anda harus lari kepada manipulasi Aljabar?
\end{latihan}

\begin{asesmen}
\textbf{Instrumen Penilaian Evaluasi Kinerja (Sub-CPMK 2.1--2.3)}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Hukum silang anomali paling ajaib bin rakus, manakala ada sebuah formula seperti $A \land (A \lor B)$, yang secara misterius mendadak dipangkas kejam musnah menjadi sisa tunggal cuma $A$ semata. Ini dinobatkan sbg Hukum:
  \begin{enumerate}
    \item De Morgan Pembongkar
    \item Idempoten Kembar
    \item Absorpsi (Penyerapan Lumat)
    \item Distributif Memecah
  \end{enumerate}
  Jawaban: (c)

  \item Menimbang proposisi majemuk: $(p \vee \neg p) \rightarrow (q \wedge \neg q)$. Di sisi kiri mencuat Tautologi $T$, di sisi kanannya kelam Kontradiksi $F$. Maka relasi raksasanya $(T \rightarrow F)$ mutlak bernilai:
  \begin{enumerate}
    \item Tautologi Abadi
    \item Kontingensi Melayang
    \item Disjungsi Inklusif
    \item Kontradiksi (Salah Total Paripurna)
  \end{enumerate}
  Jawaban: (d)

  \item Sifat magis Hukum Komutatif menjamin hak rotasi penukaran tempat duduk variabel HANYA direstui sah selama ia mengiringi instrumen:
  \begin{enumerate}
    \item Panah bersyarat Implikasi
    \item Penggabung kasta sejenis (murni kembar AND semua $\land$ atau OR semua $\lor$)
    \item Tameng Negasi dan Resolusi
    \item Derivasi Biimplikasi Mutlak
  \end{enumerate}
  Jawaban: (b)
\end{enumerate}

\textbf{B. Tugas Praktik Uji Kompetensi}
\begin{itemize}
  \item Tersaji rumus Implikasi majemuk kotor: $A \rightarrow (A \lor B)$. Manajer Anda menuntut status final. Suling peras ekspresi di atas menggunakan trik pengubahan Implikasi, terabas memakan properti Aljabar semutlak-mutlaknya hingga tak bersisa apa-apa dan putuskan apakah akhir hayat barisan tersebut dinobatkan menjadi barisan TAUTOLOGI, KONTRADIKSI, atau masih KONTINGENSI statis! Paparkan justifikasi pelangkah. 
\end{itemize}

\textbf{Rubrik Penilaian Kesahihan Penalaran Terapan}: Dapat diintip secara rinci merujuk suplemen Lampiran Buku.
\end{asesmen}

\begin{rangkuman}

\textbf{Sorotan Poin Kunci Esensial:}
\begin{itemize}
  \item \textbf{Trinitas Kasta Nilai} diklasifikasikan final ke dalam sang ratu abadi \textit{Tautologi (Selalu Benar)}, sang raja kelam \textit{Kontradiksi (Selalu Mustahil Salah)}, dan abdi dalem realitas dinamisnya si \textit{Kontingensi}.
  \item \textbf{Forensik Ekuivalensi ($\equiv$)} diinisiasi validasi sah manakala kolom pencetak akhir Tabel Kebenaran dua proposisi beda entitas ditasbihkan akurat kembar seratus persen menjiplak.
  \item Operasi turunan penyederhanaan disumbangkan secara cuma-cuma dari gudang peluru \textbf{Aljabar Proposisi} meliputi: pemotongan Idempoten/Involusi dobel, pencincang Distributif kurung silang silang gerbang, dan pembongkar gembok ganda De Morgan nan sakti.
  \item \textbf{Ekuivalensi Modifikasi} menyumbangkan sihir ilusi menembus topeng Implikasi bersyarat semu mengubahnya ringkas ke model biner $\neg p \lor q$.
  \item Segenap penatnya keringat menghafal belasan Aljabar di atas ini sejatinya \textbf{Terbayar Lunas Ratusan Juta Rupiah} kelak di dunia industri riil menyedot paksa pemotongan biaya pencetakan gerbang kabel Semikonduktor / memotong kusutnya memori verifikasi persilangan kode \textit{software If-Else bercabang}.
\end{itemize}
