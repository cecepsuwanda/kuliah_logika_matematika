\begin{aktivitas}
  \item Diskusikan dalam kelompok: Mengapa model matematis Logika Proposisi $\land, \lor, \rightarrow, \neg$ tidak sanggup memvalidasi kebenaran silogisme Aristoteles mengenai Socrates? Terangkan apa yang hilang dari logika awal tersebut.
  \item Pilihlah sebuah kalimat majemuk dari teks berita harian acak. Bedah kalimat tersebut dan temukan mana saja yang berposisi sebagai Konstan, Variabel (jika ada rujukan anonim), dan unsur verba yang menjadi Predikat.
  \item Buatlah 3 buah kalimat terbuka (Fungsi Proposisi). Tunjukkan bahwa kalimat tersebut secara absolut tidak memiliki nilai Benar maupun Salah sebelum variabelnya dieksekusi oleh konstanta dari semestanya.
  \item Definisikan fungsi $C(x, y)$ sebagai ``Mahasiswa $x$ meraih IPK lebih mantap dari $y$''. Misalkan himpunan $\mathcal{U}$ berisi seluruh prodi Informatika. Diskusikan apa yang terjadi secara logika jika Anda menyubstitusi entitas hewan (Contoh: \textit{Kucing Oren}) ke dalam variabel $x$. Hal fatal apa yang tengah Anda langgar terkait konsep Universe of Discourse?
\end{aktivitas}

\begin{latihan}
  \item Tentukan Arity (jumlah dimensi kail argumen / parameter) bagi daftar predikat di bawah ini:
  \begin{enumerate}
    \item $L(x) \equiv \text{``Tebing } x \text{ itu sangatlah curam.''}$
    \item $A(x, y) \equiv \text{``Tuan } x \text{ merayakan pernikahan dengan Nyonya } y\text{.''}$
    \item $P(a, b, c) \equiv \text{``Garis titik } a \text{ dan } b \text{ jika dihubungkan akan memotong sumbu } c\text{.''}$
  \end{enumerate}
  \item Apakah kalimat \textit{``Bilangan $x$ jika ditambahkan 10 hasilnya prima''} tergolong Proposisi Mutlak ataukah Fungsi Proposisional? Jelaskan syarat metamorfosisnya agar ia berhak disebut Proposisi berkaliber valid!
  \item Diberikan semesta $\mathcal{U} = \mathbb{Z}^+$ (Bilangan Bulat Positif). Predikat $G(n) \equiv n \text{ adalah bilangan habis dibagi nol}$. Tentukan evaluasi kebenaran untuk $n=5$, lalu diskusikan validitas klaim pada domain secara ekstrim.
  \item \textbf{Praktik Pemodelan}: Terdapat narasi, \textit{``Mahasiswi Aulia gemar makanan seblak padahal menu itu amatlah pedas''}. Misalkan Kategori semesta $\mathcal{U}$ kita sangat leluasa. Tentukan Term entitas benda, serta rumuskan setidaknya dua Predikat (satu unary, satu relasi dyadic/binary) untuk merepresentasikan naratif tersebut ke dalam format FOL.
  \item \textbf{Refleksi}: Dalam membangun sebuah sistem \textit{database} program informasi akademik, betapa signifikannya kita menetapkan spesifikasi batasan \textit{tipe data} suatu tabel. Analogi konsep apakah di ranah Logika Predikat yang sejiwa dengan pembatasan Tipe Data dalam rona Komputasi Terapan?
\end{latihan}

\begin{asesmen}
\textbf{Instrumen Evaluasi Kinerja untuk Sub-CPMK 2.1--2.2}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Elemen murni dari Logika Predikat orde pertama yang mewakili sosok entitas utuh, tidak bernilai B/S, dan bertindak serupa "Kata Benda", dipanggil dengan istilah matematis:
  \begin{enumerate}
    \item Implikasi
    \item Predikat
    \item Kuantor
    \item Term
  \end{enumerate}
  Jawaban: (d)

  \item Fungsi $F(x) = x \text{ sanggup bernapas dalam samudera}$ belum memiliki status kebenaran mutlak. Agar status validitasnya diakui secara paripurna (Sempurna \textbf{T}/\textbf{F}), $x$ tak boleh dibantarkan begitu saja, ia wajib menemui gerbang transisi:
  \begin{enumerate}
    \item Dinegasi ganda melalui Hukum De Morgan
    \item Disubstitusikan terhadap Konstanta spesifik atau dikekang dengan Jaring Kuantor.
    \item Memanipulasi Arity-nya.
    \item Digabungkan dengan operator biconditional.
  \end{enumerate}
  Jawaban: (b)

  \item Apakah sebutan bagi pemagaran deklaratif absolut mengenai "Lingkup/wadah kumpulan" nilai konstanta yang legal merasuk ke dalam relung kosong Fungsi Proposisional?
  \begin{enumerate}
    \item Subjek Partisi
    \item \textit{Universe of Discourse} (Semesta Pembicaraan)
    \item Kuantor Eksistensial
    \item Resolusi Eksternal
  \end{enumerate}
  Jawaban: (b)

  \item Kalimat \textit{"Gunung Everest mematok rekor lebih tinggi 2 kilometer ketimbang lautan terdalam Samudera Pasifik"} memuat struktur kata benda dan komparasi sifat. Apabila diusut hingga pangkal, kalimat ini paling elok diwakili oleh wujud relasi Predikat matematika bernama:
  \begin{enumerate}
    \item Predikat 1-Arity (Monadik)
    \item Predikat 2-Arity (Dyadik)
    \item Fungsi Trivial Kosong
    \item Disjungsi Silogisme
  \end{enumerate}
  Jawaban: (b)
\end{enumerate}

\begin{rangkuman}
Bab 6 meruntuhkan tembok stagnan Logika Proposisi klasik dan menuntun pergeseran metodis fundamental memasuki panggung agung Logika Predikat (Kalkulus Predikat Orde Pertama). 

\textbf{Poin Kunci:}
\begin{itemize}
  \item \textbf{Logika Proposisi Gagal membedah anatomi klausa}, sehingga ia absen mengenali rujukan antar \textit{entitas subjek}. Hal ini melatari fajar era Logika Predikat.
  \item \textbf{Term} duduk bertahta memimpin subjek kata benda. Ia tak punya sifat B/S. Ia menitis menjelma sebagai: (1) Konstanta/Individu Pasti, (2) Variabel Liar Anonim, dan (3) Output Fungsi pemetaan logis matematik.
  \item \textbf{Predikat ($P$)} menahbiskan status atribut, sifat, perilaku bagi sesosok sekumpulan \textit{Term}. Relasi parameter diukur menggunakan \textit{kantung Arity} (dari dimensi-1 berwujud properti hingga relasi silang dimensi-n).
  \item \textbf{Fungsi Proposisional / Kalimat Terbuka} dilarang keras diaudit validitas B/S-nya, hingga ia dibajak status keliarannya (Variabel digantikan \textbf{Konstanta} mutlak atau dikomando dengan Jaring Kuantifikasi makro).
  \item \textbf{Semesta Pembicaraan / Universe of Discourse ($\mathcal{U}$)} merupakan wilayah demarkasi yang melerai kesahihan argumen. Sebaris perumusan jenius bisa luruh ditelan nilai salah total jika panggung domain semestanya tergelincir bergeser.
\end{itemize}

\textbf{Kata Kunci}: Logika Pertama / FOL (\textit{First-Order Logic}), Term, Konstanta, Variabel, Predikat, Arity (Dyadik, Monadik), Fungsi Proposisional, Kalimat Terbuka Tertutup, Universe of Discourse.
\end{rangkuman}
