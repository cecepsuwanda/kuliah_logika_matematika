\begin{aktivitas}
  \item Tuliskan prekondisi dan postkondisi untuk fungsi yang menghitung nilai maksimum dari dua bilangan bulat. Verifikasi menggunakan Hoare Triple.
  \item Rancang algoritma pencarian linear pada array. Definisikan loop invariant-nya dan verfikasi ketiga syarat (inisialisasi, pemeliharaan, terminasi).
  \item Tuliskan prekondisi dan postkondisi untuk fungsi perkalian yang menggunakan penjumlahan berulang ($a \times b = \underbrace{a + a + \ldots + a}_{b \text{ kali}}$). Tentukan loop invariant dan fungsi pengurangannya.
  \item Analisis program Insertion Sort. Buktikan kebenaran parsial invarian iterasi pertama: ``Setelah iterasi $j = 1$, subarray $A[0..1]$ terurut.''
  \item Tentukan apakah program berikut memiliki kebenaran total:
  \begin{verbatim}
      x := n
      while x > 0 do
          x := x - 2
  \end{verbatim}
  Untuk $n$ genap positif dan $n$ ganjil positif, apakah program selalu berterminasi?
  \item Berikan contoh program yang memenuhi kebenaran parsial tetapi \textbf{tidak} memenuhi kebenaran total. Jelaskan mengapa.
\end{aktivitas}

% LATIHAN DAN REFLEKSI
% ============================================================

\begin{latihan}
  \item Verifikasi Hoare Triple berikut: $\{y = 3\}\ x := y + 2\ \{x = 5\}$. Gunakan aturan assignment.
  \item Diberikan program:
  \begin{verbatim}
      x := a + b
      y := x * 2
  \end{verbatim}
  Jika prekondisi $\{a = 3 \wedge b = 4\}$, tentukan postkondisi menggunakan aturan komposisi sekuensial.
  \item Rancang loop invariant dan buktikan kebenaran parsial untuk program berikut:
  \begin{verbatim}
      prod := 0; i := 0
      while i < b do
          prod := prod + a
          i := i + 1
  \end{verbatim}
  yang menghitung $a \times b$ (untuk $b \geq 0$).
  \item Buktikan terminasi program pada soal sebelumnya menggunakan fungsi pengurangan.
  \item Jelaskan mengapa ``$A$ merupakan permutasi dari masukan'' menjadi bagian penting dari postkondisi algoritma pengurutan.
  \item Tentukan loop invariant untuk algoritma Bubble Sort dan verifikasi syarat inisialisasinya.
  \item \textbf{Refleksi}: Edsger Dijkstra berkata: ``Program testing can be used to show the presence of bugs, but never to show their absence.'' Jelaskan makna pernyataan ini dalam konteks kebenaran parsial dan kebenaran total.
  \item \textbf{Refleksi}: Dalam konteks \textit{mission-critical systems} (aviasi, medis, antariksa), mengapa mengandalkan testing saja tidak cukup? Berikan contoh historis di mana kegagalan verifikasi formal mengakibatkan kerugian besar.
\end{latihan}

\begin{asesmen}
\textbf{Instrumen Evaluasi Kinerja untuk Sub-CPMK 4.7--4.9}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Dalam spesifikasi program, kondisi yang harus dipenuhi masukan sebelum eksekusi disebut:
  \begin{enumerate}
    \item Postkondisi
    \item Loop invariant
    \item Prekondisi
    \item Fungsi pengurangan
  \end{enumerate}
  Jawaban: (c)

  \item Hoare Triple $\{p\}\ S\ \{q\}$ menyatakan:
  \begin{enumerate}
    \item $p$ benar jika dan hanya jika $q$ benar
    \item Jika $p$ dipenuhi dan $S$ berterminasi, maka $q$ terpenuhi
    \item $S$ selalu berterminasi
    \item $p$ dan $q$ selalu bernilai benar
  \end{enumerate}
  Jawaban: (b)

  \item Loop invariant harus memenuhi tiga syarat, kecuali:
  \begin{enumerate}
    \item Inisialisasi (benar sebelum loop)
    \item Pemeliharaan (terjaga setiap iterasi)
    \item Optimasi (menghasilkan loop tercepat)
    \item Terminasi (bersama $\neg B$ mengimplikasikan postkondisi)
  \end{enumerate}
  Jawaban: (c)

  \item Perbedaan antara kebenaran parsial dan kebenaran total adalah:
  \begin{enumerate}
    \item Kebenaran parsial hanya berlaku untuk sebagian input
    \item Kebenaran total menjamin terminasi, kebenaran parsial tidak
    \item Kebenaran parsial tidak memerlukan loop invariant
    \item Kebenaran total tidak memerlukan postkondisi
  \end{enumerate}
  Jawaban: (b)

  \item Fungsi pengurangan (\textit{variant}) digunakan untuk membuktikan:
  \begin{enumerate}
    \item Kebenaran parsial
    \item Kebenaran postkondisi
    \item Terminasi perulangan
    \item Kebenaran prekondisi
  \end{enumerate}
  Jawaban: (c)
\end{enumerate}

\textbf{B. Essay}
\begin{enumerate}
  \item Verifikasi kebenaran total program pencarian linear: tuliskan loop invariant, buktikan ketiga syaratnya, dan buktikan terminasi menggunakan fungsi pengurangan.
  \item Jelaskan hubungan antara loop invariant dan induksi matematika. Mengapa keduanya secara struktural analog?
  \item Bandingkan secara kritis pendekatan testing dan verifikasi formal. Kapan masing-masing pendekatan lebih tepat digunakan?
\end{enumerate}

\textbf{Rubrik Penilaian}: Lihat Lampiran.
\end{asesmen}

\begin{rangkuman}
Bab ini membahas verifikasi kebenaran program sebagai aplikasi puncak dari seluruh konsep logika matematika yang telah dipelajari.

\textbf{Poin Kunci:}
\begin{itemize}
  \item \textbf{Spesifikasi program} dinyatakan melalui pasangan prekondisi-postkondisi, yang berfungsi sebagai ``kontrak'' antara pemanggil dan implementasi program.
  \item \textbf{Hoare Triple} $\{p\}\ S\ \{q\}$ memformalisasi hubungan antara prekondisi $p$, program $S$, dan postkondisi $q$. Aturan inferensi (assignment, komposisi, kondisional, perulangan) memungkinkan verifikasi deduktif.
  \item \textbf{Loop invariant} adalah predikat yang dijaga sepanjang eksekusi perulangan, diverifikasi melalui tiga syarat: inisialisasi, pemeliharaan, dan terminasi. Konsep ini analog langsung dengan induksi matematika.
  \item \textbf{Kebenaran parsial} menjamin postkondisi jika program berterminasi; \textbf{kebenaran total} menambahkan jaminan terminasi melalui fungsi pengurangan.
  \item Verifikasi formal sangat penting dalam mission-critical systems di mana kegagalan tidak dapat ditoleransi. Testing menunjukkan keberadaan bug; verifikasi formal membuktikan ketiadaannya.
\end{itemize}

\textbf{Kata Kunci}: prekondisi, postkondisi, Hoare Triple, logika Hoare, aturan assignment, loop invariant, kebenaran parsial, kebenaran total, fungsi pengurangan, terminasi, verifikasi formal
\end{rangkuman}
