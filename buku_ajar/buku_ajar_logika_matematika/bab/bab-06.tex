\begin{aktivitas}
  \item \textbf{Keterbatasan Logika Proposisi:} Diskusikan dalam kelompok mengapa logika proposisi ($\wedge, \vee, \rightarrow, \neg$) tidak dapat memvalidasi silogisme Aristoteles tentang Socrates. Jelaskan unsur apa yang hilang sehingga diperlukan logika predikat.
  \item \textbf{Term, Konstanta, Variabel:} Pilih satu kalimat dari berita atau teks. Identifikasi mana yang berperan sebagai Konstanta, mana sebagai Variabel (jika ada), dan mana yang merupakan Predikat (atribut/relasi).
  \item \textbf{Predikat dan Arity:} Buat tiga contoh: satu predikat unary $P(x)$, satu binary $Q(x,y)$, dan satu dengan tiga argumen $R(x,y,z)$. Tulis arti dalam bahasa alami dan tentukan arity masing-masing.
  \item \textbf{Fungsi Proposisional:} Buat tiga kalimat terbuka (fungsi proposisional). Tunjukkan bahwa ketiganya tidak memiliki nilai benar atau salah sebelum variabelnya disubstitusi oleh konstanta dari semesta.
  \item \textbf{Universe of Discourse:} Definisikan $C(x,y)$: ``Mahasiswa $x$ meraih IPK lebih tinggi dari $y$'' dengan semesta $\mathcal{U}$ = himpunan mahasiswa prodi Informatika. Diskusikan apa yang terjadi jika Anda mensubstitusi entitas di luar semesta (misalnya ``Kucing'') ke variabel $x$. Konsep apa yang dilanggar?
  \item \textbf{Kalimat Tertutup:} Dari satu fungsi proposisional $P(x)$, tunjukkan dua cara memperoleh proposisi tertutup: (1) substitusi konstanta spesifik, (2) pengikatan dengan kuantor (disinggung saja; detail di Bab 7).
  \item \textbf{Pemodelan Naratif:} Ambil satu paragraf pendek (dari berita atau cerita). Lakukan ekstraksi Term konstanta, daftar Predikat (unary dan binary), lalu tulis klausa proposisi formal yang merepresentasikan kalimat-kalimat dalam paragraf.
\end{aktivitas}

\begin{latihan}
  \item Tentukan arity (jumlah argumen) dari predikat berikut:
  \begin{enumerate}
    \item $L(x) \equiv \text{``Tebing } x \text{ sangat curam.''}$
    \item $A(x, y) \equiv \text{``Tuan } x \text{ menikah dengan Nyonya } y\text{.''}$
    \item $P(a, b, c) \equiv \text{``Garis melalui titik } a \text{ dan } b \text{ memotong sumbu } c\text{.''}$
  \end{enumerate}
  \item Kalimat ``Bilangan $x$ jika ditambah 10 hasilnya prima'' termasuk proposisi atau fungsi proposisional? Jelaskan syarat agar ia menjadi proposisi (kalimat tertutup).
  \item Diberikan semesta $\mathcal{U} = \mathbb{Z}^+$ dan predikat $G(n) \equiv \text{``}n \text{ habis dibagi 2''}$. Evaluasi kebenaran $G(5)$. Apa yang salah jika $\mathcal{U}$ tidak dibatasi (misalnya $n$ diganti objek bukan bilangan)?
  \item \textbf{Pemodelan:} Narasi: ``Mahasiswi Aulia gemar makanan seblak; menu itu sangat pedas.'' Dengan semesta yang Anda tentukan, identifikasi Term (konstanta) dan rumuskan minimal dua predikat: satu unary dan satu binary (relasi dyadic), lalu tulis klausa FOL-nya.
  \item \textbf{Refleksi:} Dalam sistem basis data program akademik, penetapan tipe data untuk setiap kolom sangat penting. Konsep apa dalam logika predikat yang analog dengan pembatasan tipe data tersebut? Jelaskan.
\end{latihan}

\begin{asesmen}
\textbf{Instrumen Evaluasi Kinerja (Sub-CPMK 2.1)}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Dalam logika predikat, elemen yang mewakili entitas utuh (kata benda), tidak bernilai benar/salah, dan dapat berupa nama spesifik atau variabel, disebut:
  \begin{enumerate}
    \item Predikat
    \item Kuantor
    \item Term
    \item Implikasi
  \end{enumerate}
  Jawaban: (c)

  \item Fungsi proposisional $F(x)$ belum memiliki nilai kebenaran. Agar menjadi proposisi (bernilai T atau F), variabel $x$ harus:
  \begin{enumerate}
    \item Dinegasi dengan hukum De Morgan
    \item Disubstitusi oleh konstanta spesifik atau diikat oleh kuantor
    \item Diubah arity-nya
    \item Digabung dengan biimplikasi
  \end{enumerate}
  Jawaban: (b)

  \item Himpunan nilai konstanta yang diperbolehkan untuk mensubstitusi variabel dalam fungsi proposisional disebut:
  \begin{enumerate}
    \item Partisi subjek
    \item Universe of Discourse (semesta pembicaraan)
    \item Kuantor eksistensial
    \item Resolusi
  \end{enumerate}
  Jawaban: (b)

  \item Kalimat ``Gunung Everest lebih tinggi daripada palung terdalam Samudra Pasifik'' memuat relasi antara dua objek. Dalam logika predikat, struktur ini paling tepat diwakili oleh:
  \begin{enumerate}
    \item Predikat unary (1-arity)
    \item Predikat binary (2-arity)
    \item Fungsi proposisional tanpa variabel
    \item Disjungsi
  \end{enumerate}
  Jawaban: (b)

  \item Silogisme Aristoteles (Semua manusia fana; Socrates manusia; maka Socrates fana) tidak dapat divalidasi oleh logika proposisi murni karena:
  \begin{enumerate}
    \item Logika proposisi hanya mengenal konjungsi dan disjungsi
    \item Logika proposisi memperlakukan tiap kalimat sebagai atom tanpa membedah struktur internal (subjek-predikat)
    \item Logika proposisi tidak mengenal negasi
    \item Silogisme memerlukan kuantor eksistensial saja
  \end{enumerate}
  Jawaban: (b)
\end{enumerate}

\textbf{B. Tugas Praktik}
\begin{itemize}
  \item Diberikan teks: ``Budi Setiawan bekerja di perusahaan Garuda Indonesia di kota Solo. Budi mahir menulis kode Python.'' Deskripsikan himpunan semesta pembicaraan yang wajar, daftar Term konstanta, dan transformasikan klaim dalam teks ke dalam predikat logika predikat (FOL) secara formal.
\end{itemize}

\textbf{Rubrik}: Pemetaan naratif ke FOL 50\%; akurasi definisi semesta pembicaraan 50\%. Merujuk Lampiran Buku.
\end{asesmen}

\begin{rangkuman}
Bab ini menjelaskan motivasi logika predikat (logika orde pertama, FOL) sebagai perluasan logika proposisi, serta konsep term, predikat, fungsi proposisional, dan semesta pembicaraan.

\textbf{Poin kunci:}
\begin{itemize}
  \item \textbf{Keterbatasan logika proposisi:} setiap kalimat atom tidak dibedah menjadi subjek dan predikat, sehingga validitas inferensi seperti silogisme klasik tidak terekam dalam struktur huruf proposisi semata.
  \item \textbf{Term:} ekspresi yang merujuk pada entitas; meliputi konstanta, variabel, dan hasil aplikasi fungsi terhadap term. Term tidak bernilai benar atau salah.
  \item \textbf{Predikat:} menyatakan sifat atau relasi pada satu atau lebih term; banyaknya argumen disebut \textit{arity} (misalnya unary, binary, $n$-ary).
  \item \textbf{Fungsi proposisional (kalimat terbuka):} kombinasi predikat dengan variabel bebas; nilai kebenaran baru ditentukan setelah substitusi konstanta yang sesuai semesta atau setelah pengikatan kuantor (Bab 7).
  \item \textbf{Semesta pembicaraan ($\mathcal{U}$):} himpunan objek yang diperbolehkan sebagai substitusi; tanpa $\mathcal{U}$ yang jelas, evaluasi dapat menjadi ambigu atau tidak terdefinisi secara wajar.
\end{itemize}

\textbf{Kata kunci:} logika orde pertama (FOL), term, konstanta, variabel, predikat, arity, fungsi proposisional, kalimat terbuka, kalimat tertutup, semesta pembicaraan (\textit{universe of discourse}).
\end{rangkuman}
