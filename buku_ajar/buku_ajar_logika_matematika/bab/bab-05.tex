% AKTIVITAS PEMBELAJARAN
% ============================================================

\begin{aktivitas}
  \item \textbf{Validitas Argumen dan Tautologi:} Pilih satu argumen pendek dari berita atau iklan. Tulis premis dan konklusinya dalam notasi proposisional. Verifikasi validitas dengan mengecek apakah bentuk $(P_1 \wedge P_2 \wedge \ldots) \rightarrow Q$ tautologi (tabel kebenaran).
  \item \textbf{Modus Ponens dan Modus Tollens:} Berikan masing-masing satu contoh kehidupan sehari-hari yang memenuhi skema Modus Ponens dan satu contoh Modus Tollens. Tulis dalam bentuk premis–konklusi.
  \item \textbf{Silogisme Hipotetis:} Tulis argumen ``Jika hujan maka jalanan licin. Jika jalanan licin maka kendaraan melambat'' ke notasi aljabar proposisi. Buktikan validitas Silogisme Hipotetis melalui tabel kebenaran (8 baris).
  \item \textbf{Aturan Konjungsi, Penambahan, Simplifikasi:} Diskusikan dalam kelompok: dalam pembuktian berantai, kapan aturan Konjungsi dipakai? Kapan Penambahan (Addition) dan kapan Simplifikasi? Berikan satu contoh singkat untuk masing-masing.
  \item \textbf{Aturan Resolusi:} Diberikan dua klausa $(p \vee q)$ dan $(\neg p \vee r)$. Tentukan resolvent-nya. Diskusikan mengapa aturan ini menjadi dasar inferensi dalam sistem seperti PROLOG.
  \item \textbf{Fallacy:} Cari di media (debat, iklan, atau berita) minimal satu contoh yang menyerupai Fallacy of Affirming the Consequent dan satu contoh Denying the Antecedent. Tulis transkrip singkat dan jelaskan kelemahan logikanya.
  \item \textbf{Bukti Formal Berantai:} Dari tiga premis yang diberikan dosen atau dari buku, susun bukti formal berantai menuju konklusi yang ditentukan. Setiap langkah harus disertai justifikasi (nama aturan dan nomor baris).
\end{aktivitas}

% LATIHAN DAN REFLEKSI
% ============================================================

\begin{latihan}
  \item Tentukan aturan inferensi yang digunakan pada argumen berikut:
  \begin{enumerate}
    \item Premis 1: ``Jika Anda kaya, maka Anda bebas.'' Premis 2: ``Anda tidak bebas.'' Konklusi: ``Maka Anda tidak kaya.''
    \item Premis 1: ``Ikan paus terdampar lapar atau ia menari.'' Premis 2: ``Dibuktikan paus tidak lapar.'' Konklusi: ``Maka paus sedang menari.''
  \end{enumerate}
  \item Diberikan premis: (1) $(P \vee Q) \rightarrow R$, (2) $S \rightarrow \neg R$, (3) $S$. Susun bukti formal berantai (tabel baris dengan justifikasi) untuk mencapai konklusi $\neg P$.
  \item Diberikan dua klausa $(p \vee q)$ dan $(\neg p \vee r)$. Tuliskan resolvent-nya dan jelaskan aturan yang dipakai.
  \item Jelaskan perbedaan antara Modus Ponens $(p \rightarrow q, p \vdash q)$ dengan pola invalid ``$p \rightarrow q$, $q$, maka $p$'' (Fallacy of Affirming the Consequent). Berikan satu contoh yang menunjukkan bahwa pola kedua tidak sah.
  \item \textbf{Refleksi:} Mengapa Modus Ponens terasa natural dalam penalaran sehari-hari? Mengapa Fallacy of Denying the Antecedent sering menyesatkan? Kapan Anda memilih pengecekan validitas via tautologi dan kapan via aturan inferensi?
\end{latihan}

\begin{asesmen}
\textbf{Instrumen Evaluasi Kinerja (Sub-CPMK 1.3)}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Suatu argumen dengan premis $P_1, \ldots, P_n$ dan konklusi $Q$ disebut \textbf{valid} jika dan hanya jika:
  \begin{enumerate}
    \item Semua premis bernilai benar
    \item $(P_1 \wedge P_2 \wedge \ldots \wedge P_n) \rightarrow Q$ merupakan tautologi
    \item $Q$ bernilai benar
    \item $P_1 \vee P_2 \vee \ldots \vee P_n$ bernilai benar
  \end{enumerate}
  Jawaban: (b)

  \item Argumen dengan premis $(p \vee q)$ dan $\neg p$ yang menghasilkan konklusi $q$ menggunakan aturan:
  \begin{enumerate}
    \item Silogisme Disjungtif
    \item Silogisme Hipotetis
    \item Resolusi
    \item Simplifikasi
  \end{enumerate}
  Jawaban: (a)

  \item Dari implikasi $(p \rightarrow q)$, seseorang keliru menyimpulkan $p$ benar karena $q$ benar. Kesalahan ini disebut:
  \begin{enumerate}
    \item Fallacy of Denying the Antecedent
    \item Fallacy of Affirming the Consequent
    \item Modus Tollens
    \item Silogisme Hipotetis
  \end{enumerate}
  Jawaban: (b)

  \item Aturan Simplifikasi berlaku pada bentuk yang memuat operator:
  \begin{enumerate}
    \item Implikasi ($\rightarrow$)
    \item Konjungsi ($\wedge$)
    \item Disjungsi ($\vee$)
    \item Biimplikasi ($\leftrightarrow$)
  \end{enumerate}
  Jawaban: (b)

  \item Dari premis $p \rightarrow q$ dan $q \rightarrow r$, aturan Silogisme Hipotetis menyimpulkan:
  \begin{enumerate}
    \item $p \wedge r$
    \item $p \rightarrow r$
    \item $q \rightarrow p$
    \item $\neg q$
  \end{enumerate}
  Jawaban: (b)

  \item Mekanisme inferensi yang menggabungkan dua klausa disjungsi dengan literal komplementer (misalnya $p \vee q$ dan $\neg p \vee r$) menghasilkan resolvent. Teknik ini menjadi dasar dalam:
  \begin{enumerate}
    \item Hanya Modus Ponens
    \item Aturan Resolusi (sistem seperti PROLOG)
    \item Fallacy of Affirming the Consequent
    \item Simplifikasi
  \end{enumerate}
  Jawaban: (b)
\end{enumerate}

\textbf{B. Tugas Praktik}
\begin{itemize}
  \item Diberikan premis: ``Jika saya tidak melunasi pajak, mobil saya diderek.'' Fakta: mobil tidak diderek (aman di garasi). Formulasikan dalam notasi logika proposisional, gunakan aturan inferensi yang tepat untuk menyimpulkan status pajak, dan sebutkan nama aturan yang digunakan.
\end{itemize}

\textbf{Rubrik Penilaian}: Merujuk Lampiran Buku.
\end{asesmen}

\begin{rangkuman}
Bab 5 membahas penarikan kesimpulan yang sah melalui aturan inferensi dan cara mengenali pola yang tidak sah (\textit{fallacy}).

\textbf{Poin-poin utama:}
\begin{itemize}
  \item \textbf{Argumen valid} jika dan hanya jika $(P_1 \land P_2 \land \ldots \land P_n) \rightarrow Q$ merupakan tautologi.
  \item \textbf{Modus Ponens} menurunkan konsekuen dari implikasi yang benar dan anteseden yang benar. \textbf{Modus Tollens} menurunkan negasi anteseden dari implikasi yang benar dan negasi konsekuen (sejalan dengan kontraposisi).
  \item \textbf{Silogisme hipotetis} menghubungkan rantai implikasi; \textbf{silogisme disjungtif} memilih salah satu cabang disjungsi ketika cabang lain ditetapkan salah.
  \item \textbf{Aturan resolusi} menjadi landasan inferensi pada basis pengetahuan dalam sistem pakar dan bahasa logika seperti PROLOG.
  \item \textbf{Fallacy formal} meliputi \textit{affirming the consequent} (menyimpulkan anteseden dari konsekuen) dan \textit{denying the antecedent} (menyimpulkan negasi konsekuen dari negasi anteseden).
  \item \textbf{Deduksi formal} mensyaratkan justifikasi pada setiap langkah pembuktian menuju konklusi.
\end{itemize}

\textbf{Kata kunci}: Inferensi, validitas argumen, Modus Ponens, Modus Tollens, silogisme, resolusi, fallacy, bukti formal berantai, tautologi argumen.
\end{rangkuman}
