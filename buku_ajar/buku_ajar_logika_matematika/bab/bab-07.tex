% AKTIVITAS PEMBELAJARAN
% ============================================================

\begin{aktivitas}
  \item \textbf{Semesta dan nilai kebenaran:} Tentukan nilai kebenaran $\forall x \, (x^2 \ge x)$ jika $\mathcal{U} = \mathbb{Z}$. Ulangi untuk $\mathcal{U} = \mathbb{R}$. Berikan counterexample jika pernyataan salah.
  \item \textbf{Kuantor eksistensial:} Buat minimal tiga pernyataan berkuantor eksistensial dalam konteks sistem basis data mahasiswa (misalnya: ada mahasiswa yang..., ada mata kuliah yang...).
  \item \textbf{Universal dan implikasi:} Translasi kalimat ``Setiap burung bisa terbang'' ke logika predikat dengan semesta = semua hewan. Gunakan predikat $B(x)$ dan $F(x)$. Jelaskan mengapa $\forall x \, (B(x) \rightarrow F(x))$ dipilih, bukan $\forall x \, (B(x) \land F(x))$.
  \item \textbf{Kuantor bersarang:} Diskusikan mengapa $\exists y \, \forall x \, P(x,y)$ lebih kuat (lebih sulit dibuktikan) daripada $\forall x \, \exists y \, P(x,y)$. Berikan contoh dengan $P(x,y)$ = ``$x$ mencintai $y$''.
  \item \textbf{Negasi berkuantor:} Lakukan negasi bertahap pada: ``Semua mahasiswa baru wajib mengikuti orientasi atau mengumpulkan surat keterangan sakit.'' Gunakan hukum De Morgan untuk kuantor dan operator $\vee$.
  \item \textbf{Translasi:} Dari kamus translasi (section 7-7), pilih dua frasa bahasa Indonesia berbeda dan tulis rumus FOL-nya beserta pola kuantor-operator yang digunakan.
  \item \textbf{Evaluasi pada domain terhingga:} Jika $\mathcal{U} = \{1, 2, 3\}$, jabarkan $\forall x \, P(x)$ dan $\exists x \, P(x)$ menjadi bentuk konjungsi/disjungsi tanpa simbol kuantor.
\end{aktivitas}

% LATIHAN DAN REFLEKSI
% ============================================================

\begin{latihan}
  \item Misalkan $M(x)$: ``$x$ adalah mahasiswa'' dan $L(x)$: ``$x$ lulus kalkulus''. Translasi ke logika predikat:
  \begin{enumerate}
    \item Setiap mahasiswa lulus kalkulus.
    \item Ada mahasiswa yang lulus kalkulus.
    \item Tidak ada mahasiswa yang lulus kalkulus.
    \item Beberapa mahasiswa tidak lulus kalkulus.
  \end{enumerate}
  \item Tentukan negasi dari $\exists x \, \forall y \, (x + y = 0)$ dan sederhanakan hingga tidak ada negasi di depan kuantor.
  \item Jika $\mathcal{U} = \{1, 2, 3\}$, jabarkan $\forall x \, \exists y \, (x < y)$ menjadi bentuk konjungsi dan disjungsi tanpa simbol kuantor.
  \item Jelaskan mengapa kuantor universal ($\forall$) lazim dipasangkan dengan implikasi ($\rightarrow$) dan kuantor eksistensial ($\exists$) dengan konjungsi ($\land$). Berikan satu contoh kesalahan translasi jika pasangan tersebut dibalik.
  \item \textbf{Refleksi:} Berikan contoh kalimat dalam bahasa sehari-hari yang ambigu karena scope kuantitas (semua vs sebagian) tidak jelas. Bagaimana logika predikat mengatasi ambiguitas tersebut?
\end{latihan}

\begin{asesmen}
\textbf{Instrumen Evaluasi Kinerja (Sub-CPMK 2.2)}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Translasi formal yang tepat untuk ``Setiap burung ($B$) bisa terbang ($F$)'' adalah:
  \begin{enumerate}
    \item $\forall x \, (B(x) \land F(x))$
    \item $\forall x \, (B(x) \rightarrow F(x))$
    \item $\exists x \, (B(x) \rightarrow F(x))$
    \item $\forall x \, (F(x) \rightarrow B(x))$
  \end{enumerate}
  Jawaban: (b)

  \item Jika $\mathcal{U}$ = himpunan semua hewan dan $B(x)$ = ``$x$ bisa berbicara'', terjemahan yang tepat untuk ``Ada hewan yang bisa berbicara'' adalah:
  \begin{enumerate}
    \item $\exists x \, B(x)$
    \item $\forall x \, B(x)$
    \item $\exists x \, \neg B(x)$
    \item $\forall x \, \neg B(x)$
  \end{enumerate}
  Jawaban: (a)

  \item Hukum De Morgan untuk kuantor: $\neg \forall x \, P(x)$ ekuivalen dengan:
  \begin{enumerate}
    \item $\forall x \, \neg P(x)$
    \item $\exists x \, \neg P(x)$
    \item $\neg \exists x \, P(x)$
    \item $\exists x \, P(x)$
  \end{enumerate}
  Jawaban: (b)

  \item Pernyataan $\forall x \, \exists y \, P(x,y)$ bermakna:
  \begin{enumerate}
    \item Ada satu $y$ yang sama untuk semua $x$.
    \item Untuk setiap $x$ terdapat (minimal satu) $y$ yang bergantung pada $x$ sehingga $P(x,y)$ benar.
    \item Setiap pasangan $(x,y)$ memenuhi $P(x,y)$.
    \item Pernyataan selalu benar di semesta mana pun.
  \end{enumerate}
  Jawaban: (b)
\end{enumerate}

\textbf{B. Tugas Praktik}
\begin{enumerate}
  \item Dengan $S(x)$ = mahasiswa, $K(x)$ = komputer, $P(x,y)$ = $x$ memiliki $y$, translasikan ke logika predikat: ``Setiap mahasiswa memiliki sekurang-kurangnya satu komputer.''
  \item Jabarkan proses negasi (De Morgan untuk kuantor) hingga ke dalam scope terdalam untuk: $\neg \big[ \exists x \, \forall y \, (P(x) \rightarrow Q(x,y)) \big]$.
\end{enumerate}

\textbf{Rubrik}: Merujuk Lampiran Buku.
\end{asesmen}

% RANGKUMAN

\begin{rangkuman}
Bab 7 membahas kuantor universal dan eksistensial sebagai perangkat utama dalam logika predikat untuk menutup variabel bebas dan menyatakan cakupan objek pada suatu domain.

\textbf{Poin Kunci:}
\begin{itemize}
  \item \textbf{Semesta pembicaraan ($\mathcal{U}$)} menentukan himpunan nilai yang diperbolehkan untuk variabel, sehingga memengaruhi nilai kebenaran proposisi berkuantor.
  \item \textbf{Kuantor universal ($\forall$)} menyatakan ``setiap/semua'' dan, dalam translasi umum, dipadukan dengan implikasi untuk pola ``setiap $P$ adalah $Q$''.
  \item \textbf{Kuantor eksistensial ($\exists$)} menyatakan ``ada/sebagian'' dan, dalam translasi umum, dipadukan dengan konjungsi untuk pola ``ada $P$ yang $Q$''.
  \item \textbf{Urutan kuantor} memengaruhi makna; bentuk $\forall x \, \exists y \, P(x,y)$ tidak ekuivalen dengan $\exists y \, \forall x \, P(x,y)$.
  \item \textbf{Negasi kuantor} mengikuti hukum De Morgan: $\neg \forall x \, P(x) \equiv \exists x \, \neg P(x)$ dan $\neg \exists x \, P(x) \equiv \forall x \, \neg P(x)$.
\end{itemize}

\textbf{Kata Kunci}: semesta pembicaraan, kuantor universal, kuantor eksistensial, implikasi, konjungsi, kuantor bersarang, negasi kuantor, \textit{witness}, \textit{counterexample}.
\end{rangkuman}
