\begin{aktivitas}
  \item \textbf{Hoare Triple:} Tuliskan prekondisi dan postkondisi untuk fungsi yang menghitung nilai maksimum dari dua bilangan bulat. Verifikasi dengan Hoare Triple.
  \item \textbf{Loop invariant:} Rancang algoritma pencarian linear pada array. Definisikan loop invariant dan verifikasi ketiga syarat (inisialisasi, pemeliharaan, terminasi).
  \item \textbf{Perkalian berulang:} Tuliskan prekondisi dan postkondisi untuk fungsi perkalian dengan penjumlahan berulang ($a \times b = a + a + \cdots + a$ sebanyak $b$ kali). Tentukan loop invariant dan fungsi pengurangan.
  \item \textbf{Insertion Sort:} Analisis program Insertion Sort. Buktikan kebenaran parsial invarian iterasi pertama: ``Setelah iterasi $j = 1$, subarray $A[0..1]$ terurut.''
  \item \textbf{Terminasi:} Untuk program \texttt{while $x > 0$ do $x := x - 2$} dengan $x := n$ awal, tentukan apakah kebenaran total berlaku. Untuk $n$ genap positif dan $n$ ganjil positif, apakah program selalu berterminasi?
  \item \textbf{Parsial vs total:} Berikan contoh program yang memenuhi kebenaran parsial tetapi tidak memenuhi kebenaran total. Jelaskan alasannya.
\end{aktivitas}

% ============================================================

\begin{latihan}
  \item Verifikasi Hoare Triple: $\{y = 3\}\ x := y + 2\ \{x = 5\}$ menggunakan aturan assignment.
  \item Diberikan program: \texttt{$x := a + b$; $y := x * 2$}. Jika prekondisi $\{a = 3 \wedge b = 4\}$, tentukan postkondisi dengan aturan komposisi sekuensial.
  \item Rancang loop invariant dan buktikan kebenaran parsial untuk program yang menghitung $a \times b$ ($b \geq 0$): \texttt{prod := 0; i := 0; while i < b do prod := prod + a; i := i + 1}.
  \item Buktikan terminasi program pada soal sebelumnya menggunakan fungsi pengurangan.
  \item Jelaskan mengapa ``$A$ merupakan permutasi dari masukan'' menjadi bagian penting dari postkondisi algoritma pengurutan.
  \item Tentukan loop invariant untuk algoritma Bubble Sort dan verifikasi syarat inisialisasinya.
  \item \textbf{Refleksi:} (a) Edsger Dijkstra: ``Program testing can show the presence of bugs, but never their absence.'' Jelaskan dalam konteks kebenaran parsial dan kebenaran total. (b) Dalam sistem mission-critical (aviasi, medis, antariksa), mengapa testing saja tidak cukup? Berikan contoh historis kegagalan verifikasi formal.
\end{latihan}

\begin{asesmen}
\textbf{Instrumen Evaluasi Kinerja (Sub-CPMK 4.3)}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Kondisi yang harus dipenuhi masukan sebelum eksekusi program disebut:
  \begin{enumerate}
    \item Postkondisi
    \item Loop invariant
    \item Prekondisi
    \item Fungsi pengurangan
  \end{enumerate}
  Jawaban: (c)

  \item Hoare Triple $\{p\}\ S\ \{q\}$ menyatakan:
  \begin{enumerate}
    \item $p$ benar jika dan hanya jika $q$ benar
    \item Jika $p$ dipenuhi dan $S$ berterminasi, maka $q$ terpenuhi
    \item $S$ selalu berterminasi
    \item $p$ dan $q$ selalu benar
  \end{enumerate}
  Jawaban: (b)

  \item Loop invariant harus memenuhi tiga syarat, kecuali:
  \begin{enumerate}
    \item Inisialisasi (benar sebelum loop)
    \item Pemeliharaan (terjaga setiap iterasi)
    \item Optimasi (menghasilkan loop tercepat)
    \item Terminasi (bersama $\neg B$ mengimplikasikan postkondisi)
  \end{enumerate}
  Jawaban: (c)

  \item Perbedaan kebenaran parsial dan kebenaran total adalah:
  \begin{enumerate}
    \item Kebenaran parsial hanya berlaku untuk sebagian input
    \item Kebenaran total menjamin terminasi; kebenaran parsial tidak
    \item Kebenaran parsial tidak memerlukan loop invariant
    \item Kebenaran total tidak memerlukan postkondisi
  \end{enumerate}
  Jawaban: (b)

  \item Fungsi pengurangan (variant) digunakan untuk membuktikan:
  \begin{enumerate}
    \item Kebenaran parsial
    \item Kebenaran postkondisi
    \item Terminasi perulangan
    \item Kebenaran prekondisi
  \end{enumerate}
  Jawaban: (c)
\end{enumerate}

\textbf{B. Tugas Praktik}
\begin{enumerate}
  \item Verifikasi kebenaran total program pencarian linear: tuliskan loop invariant, buktikan ketiga syaratnya, dan buktikan terminasi dengan fungsi pengurangan.
  \item Jelaskan hubungan antara loop invariant dan induksi matematika. Mengapa keduanya secara struktural analog?
  \item Bandingkan pendekatan testing dan verifikasi formal. Kapan masing-masing lebih tepat digunakan?
\end{enumerate}

\textbf{Rubrik}: Merujuk Lampiran Buku.
\end{asesmen}

\begin{rangkuman}
Bab ini membahas verifikasi kebenaran program sebagai penerapan yang mengintegrasikan konsep logika matematika yang telah dipelajari.

\textbf{Poin kunci:}
\begin{itemize}
  \item \textbf{Spesifikasi program} dinyatakan melalui pasangan prekondisi-postkondisi, yang berfungsi sebagai ``kontrak'' antara pemanggil dan implementasi program.
  \item \textbf{Hoare Triple} $\{p\}\ S\ \{q\}$ memformalisasi hubungan antara prekondisi $p$, program $S$, dan postkondisi $q$. Aturan inferensi (assignment, komposisi, kondisional, perulangan) memungkinkan verifikasi deduktif.
  \item \textbf{Loop invariant} adalah predikat yang dijaga sepanjang eksekusi perulangan, diverifikasi melalui tiga syarat: inisialisasi, pemeliharaan, dan terminasi. Konsep ini analog langsung dengan induksi matematika.
  \item \textbf{Kebenaran parsial} menjamin postkondisi jika program berterminasi; \textbf{kebenaran total} menambahkan jaminan terminasi melalui fungsi pengurangan.
  \item Verifikasi formal menjadi prioritas dalam mission-critical systems di mana kegagalan tidak dapat ditoleransi. Testing menunjukkan keberadaan bug; verifikasi formal membuktikan ketiadaannya.
\end{itemize}

\textbf{Kata kunci:} prekondisi, postkondisi, Hoare Triple, logika Hoare, aturan assignment, loop invariant, kebenaran parsial, kebenaran total, fungsi pengurangan, terminasi, verifikasi formal
\end{rangkuman}
