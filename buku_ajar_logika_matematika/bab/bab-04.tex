\begin{asesmen}
\textbf{Instrumen Penilaian untuk Sub-CPMK 2.1--2.3}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Dua proposisi $P$ dan $Q$ ekuivalen secara logika jika dan hanya jika:
  \begin{enumerate}
    \item $P \wedge Q$ tautologi
    \item $P \leftrightarrow Q$ tautologi
    \item $P \vee Q$ kontradiksi
    \item $P \rightarrow Q$ kontingensi
  \end{enumerate}
  \item Hukum De Morgan: $\neg(p \vee q)$ ekuivalen dengan:
  \begin{enumerate}
    \item $\neg p \vee \neg q$
    \item $\neg p \wedge \neg q$
    \item $p \wedge q$
    \item $p \vee q$
  \end{enumerate}
  \item Proposisi yang selalu bernilai salah untuk setiap interpretasi disebut:
  \begin{enumerate}
    \item Tautologi
    \item Kontradiksi
    \item Kontingensi
    \item Ekuivalen
  \end{enumerate}
  \item Hukum distributif: $p \wedge (q \vee r)$ ekuivalen dengan:
  \begin{enumerate}
    \item $(p \wedge q) \vee (p \wedge r)$
    \item $(p \vee q) \wedge (p \vee r)$
    \item $p \wedge q \wedge r$
    \item $(p \wedge q) \vee r$
  \end{enumerate}
  \item Translasi yang benar untuk ``Saya lulus hanya jika saya belajar'' ($p$: saya lulus, $q$: saya belajar) adalah:
  \begin{enumerate}
    \item $q \rightarrow p$
    \item $p \rightarrow q$
    \item $p \wedge q$
    \item $p \leftrightarrow q$
  \end{enumerate}
\end{enumerate}

\textbf{B. Tugas Praktik}
\begin{itemize}
  \item Translasikan dua kalimat bahasa Indonesia ke simbol logika (identifikasi proposisi atomik dan operator). Verifikasi dengan tabel kebenaran.
  \item Tulis program Python singkat yang memeriksa apakah suatu formula (untuk 2 variabel) adalah tautologi, kontradiksi, atau kontingensi.
  \item Sederhanakan ekspresi $\neg(p \rightarrow q) \vee (p \wedge q)$ menggunakan hukum aljabar proposisi. Tuliskan setiap langkah.
\end{itemize}

\begin{rangkuman}
Bab ini membahas ekuivalensi logika, tautologi, dan translasi pernyataan ke simbol logika.

\textbf{Poin Kunci:}
\begin{itemize}
  \item Dua formula $P$ dan $Q$ ekuivalen ($P \equiv Q$) jika $P \leftrightarrow Q$ adalah tautologi; hukum-hukum aljabar proposisi (komutatif, asosiatif, distributif, De Morgan, absorpsi, identitas, komplemen, idempoten) menyediakan aturan transformasi.
  \item Tautologi (selalu benar), kontradiksi (selalu salah), dan kontingensi (tergantung input) menentukan sifat formula logika.
  \item Translasi bahasa alami ke simbol logika membutuhkan identifikasi proposisi atomik, pemilihan operator yang tepat, dan penanganan kata kunci seperti ``hanya jika'', ``kecuali'', dan ``jika dan hanya jika''.
  \item Penyederhanaan ekspresi logika menggunakan hukum-hukum aljabar memungkinkan optimasi kondisi program dan verifikasi aturan bisnis.
  \item Penerapan: penyederhanaan kondisi program if/else, optimasi query Boolean, dan verifikasi spesifikasi sistem.
\end{itemize}

\textbf{Kata Kunci}: ekuivalensi logika, hukum De Morgan, hukum distributif, tautologi, kontradiksi, kontingensi, translasi, penyederhanaan, aljabar proposisi
\end{rangkuman}
