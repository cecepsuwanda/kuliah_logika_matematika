\section{Representasi Kalimat dan Variabel Bebas/Terikat}

Section ini membahas bagaimana merepresentasikan pernyataan bahasa alami ke dalam kalkulus predikat, serta konsep penting variabel bebas dan terikat.

\subsection{Representasi Bahasa Alami dalam Kalkulus Predikat}

Salah satu kemampuan penting dalam logika predikat adalah menerjemahkan pernyataan dalam bahasa alami ke notasi formal kalkulus predikat.

\begin{contoh}
Berikut pola-pola representasi yang umum:

\begin{enumerate}
  \item ``\textbf{Ada} apel berwarna merah'' \\
  $(\exists x)\, (\text{Apel}(x) \wedge \text{Merah}(x))$

  \item ``\textbf{Semua} apel berwarna merah'' \\
  $(\forall x)\, (\text{Apel}(x) \rightarrow \text{Merah}(x))$

  \item ``\textbf{Setiap} orang mencintai \textbf{seseorang}'' \\
  $(\forall x)\, (\exists y)\, \text{LOVES}(x, y)$

  \item ``Ani dicintai \textbf{banyak} orang'' \\
  $(\forall x)\, \text{LOVES}(x, \text{Ani})$
\end{enumerate}
\end{contoh}

\begin{catatan}
Perhatikan pola penting berikut:
\begin{itemize}
  \item ``\textbf{Semua} A adalah B'' $\mapsto$ $(\forall x)\,(\text{A}(x) \rightarrow \text{B}(x))$ \quad (menggunakan $\rightarrow$)
  \item ``\textbf{Ada} A yang B'' $\mapsto$ $(\exists x)\,(\text{A}(x) \wedge \text{B}(x))$ \quad (menggunakan $\wedge$)
\end{itemize}
Kesalahan umum: menggunakan $\wedge$ untuk ``semua'' atau $\rightarrow$ untuk ``ada''.
\end{catatan}

\begin{contoh}
Pernyataan yang lebih kompleks:

\begin{enumerate}
  \item ``\textbf{Semua} apel berwarna merah terasa manis'' \\
  $(\forall x)\, (\text{Apel}(x) \wedge \text{Merah}(x) \rightarrow \text{Manis}(x))$ \\
  atau secara ekuivalen: \\
  $(\forall x)\, (\text{Apel}(x) \rightarrow (\text{Merah}(x) \rightarrow \text{Manis}(x)))$

  \item ``\textbf{Tidak semua} apel berwarna merah terasa manis'' \\
  $\neg[(\forall x)\, (\text{Apel}(x) \wedge \text{Merah}(x) \rightarrow \text{Manis}(x))]$ \\
  yang ekuivalen dengan: \\
  $(\exists x)\, (\text{Apel}(x) \wedge \text{Merah}(x) \wedge \neg\text{Manis}(x))$
\end{enumerate}
\end{contoh}

\subsection{Variabel Bebas dan Variabel Terikat}

\begin{konsep}
\begin{enumerate}
  \item Suatu variabel dikatakan \logic{terikat} (\emph{bound}) dalam sebuah ekspresi jika variabel tersebut berada dalam lingkup suatu kuantifier ($\forall$ atau $\exists$).
  \item Sebaliknya, variabel dikatakan \logic{bebas} (\emph{free}) jika tidak berada dalam lingkup kuantifier manapun.
\end{enumerate}
\end{konsep}

\begin{contoh}
Perhatikan ekspresi:
$$(\forall x)\,[p(x, y) \wedge (\exists y)\, q(y, z)]$$

Analisis variabel:
\begin{itemize}
  \item $x$ pada $p(x, y)$: \textbf{terikat} (oleh $\forall x$)
  \item $y$ pada $p(x, y)$: \textbf{bebas} (tidak berada dalam lingkup kuantifier $\exists y$)
  \item $y$ pada $q(y, z)$: \textbf{terikat} (oleh $\exists y$)
  \item $z$ pada $q(y, z)$: \textbf{bebas}
\end{itemize}
\end{contoh}

\begin{konsep}
Kemunculan variabel terikat dipengaruhi oleh kuantifier yang \textbf{paling dekat}. Jika ada dua kuantifier untuk variabel yang sama secara bersarang, variabel diikat oleh kuantifier terdalam.
\end{konsep}

\begin{contoh}
Dalam ekspresi:
$$(\forall x)\,[p(x) \vee (\exists x)\,(\forall y)\, r(x, y)]$$
\begin{itemize}
  \item Variabel $x$ pada $p(x)$ dipengaruhi kuantifier $\forall x$ (luar)
  \item Variabel $x$ pada $r(x, y)$ dipengaruhi kuantifier $\exists x$ (lebih dekat)
  \item Variabel $y$ pada $r(x, y)$ dipengaruhi kuantifier $\forall y$
\end{itemize}

Perbedaan variabel bebas dan terikat:
\begin{itemize}
  \item \textbf{Variabel Bebas}: nilainya diberikan oleh interpretasi
  \item \textbf{Variabel Terikat}: nilainya terbatas dari interpretasi yang diberikan (dikendalikan oleh kuantifier)
\end{itemize}
\end{contoh}

\subsection{Kalimat Tertutup}

\begin{konsep}
Sebuah kalimat dikatakan \logic{tertutup} (\emph{closed sentence}) jika tidak mempunyai kemunculan bebas dari variabel-variabelnya. Dengan kata lain, semua variabel dalam kalimat tersebut terikat oleh kuantifier.
\end{konsep}

\begin{contoh}
\begin{enumerate}
  \item $(\forall x)\, (\exists y)\, p(x, y)$ adalah kalimat \textbf{tertutup} (semua variabel terikat)
  \item $(\forall x)\, p(x, y)$ \textbf{bukan} kalimat tertutup (variabel $y$ bebas)
\end{enumerate}
\end{contoh}

\subsection{Simbol Bebas}

\begin{konsep}
\logic{Simbol bebas} dari ekspresi $A$ meliputi:
\begin{itemize}
  \item Variabel-variabel bebas
  \item Semua konstanta
  \item Semua simbol fungsi
  \item Semua simbol predikat
\end{itemize}
yang terdapat dalam ekspresi $A$. Simbol-simbol inilah yang harus diberi nilai oleh suatu interpretasi.
\end{konsep}
