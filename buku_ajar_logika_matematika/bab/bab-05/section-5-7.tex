\section{Interpretasi dan Aturan Semantik}

Section ini membahas bagaimana kalimat kalkulus predikat memperoleh makna melalui interpretasi, serta aturan semantik yang menentukan nilai kebenaran kalimat berkuantifier.

\subsection{Interpretasi}

\begin{konsep}
Misalkan $D$ adalah sebarang himpunan tak kosong (disebut \logic{domain}). Sebuah \logic{interpretasi} $I$ dalam domain $D$ memberikan nilai pada setiap simbol bebas dalam kalimat, dengan aturan:
\begin{itemize}
  \item Untuk setiap konstanta $a$: elemen $a_I$ dari $D$
  \item Untuk setiap variabel bebas $x$: elemen $x_I$ dari $D$
  \item Untuk setiap simbol fungsi $f$ dengan arity $= n$: fungsi $f_I(d_1, d_2, \ldots, d_n)$ dimana argumen dan nilainya merupakan elemen $D$
  \item Untuk setiap simbol predikat $p$ dengan arity $= n$: relasi $p_I(d_1, d_2, \ldots, d_n)$ yang bernilai TRUE atau FALSE
\end{itemize}
\end{konsep}

\subsection{Arti Kalimat Berdasarkan Interpretasi}

Arti kalimat ditentukan oleh interpretasi yang diberikan. Karena kalkulus predikat mengandung konsep objek, interpretasi harus mendefinisikan domain (himpunan objek) yang memberi arti pada term.

\begin{contoh}
Misalkan ada kalimat tertutup:
$$A : \text{IF}\; (\forall x)(\exists y)\, p(x, y) \;\text{THEN}\; p(a, f(a))$$

Interpretasi $I$ untuk kalimat $A$ harus:
\begin{itemize}
  \item Mendefinisikan domain $D$
  \item Memberikan nilai untuk simbol bebas: konstanta $a$, simbol fungsi $f$, simbol predikat $p$
\end{itemize}

\textbf{Interpretasi $I$}: Domain $D$ = bilangan integer positif, dimana:
\begin{itemize}
  \item $a = 0$
  \item $p$ = relasi ``lebih besar'': $p_I(d_1, d_2) = (d_1 > d_2)$
  \item $f$ = fungsi suksesor: $f_I(d) = d + 1$
\end{itemize}

Berdasarkan $I$, kalimat $A$ diartikan sebagai:\\
``IF untuk setiap integer $x$ ada integer $y$ sedemikian sehingga $x > y$, THEN $0 > 0 + 1$''
\end{contoh}

\begin{contoh}
\textbf{Interpretasi $J$}: Domain = bilangan integer positif, dimana:
\begin{itemize}
  \item $a = 0$
  \item $p$ = relasi ketidaksamaan: $p_J(d_1, d_2) = (d_1 \neq d_2)$
  \item $f$ = fungsi predeseor: $f_J(d) = d - 1$
\end{itemize}

Berdasarkan $J$, kalimat $A$ diartikan sebagai:\\
``IF untuk setiap integer $x$ ada integer $y$ sedemikian sehingga $x \neq y$, THEN $0 \neq 0 - 1$''
\end{contoh}

\subsection{Interpretasi yang Diperluas}

\begin{konsep}
Misalkan $I$ adalah interpretasi yang mencakup domain $D$. Untuk sembarang variabel $x$ dan elemen $d$ dari $D$, \logic{interpretasi yang diperluas}:
$$\langle x \leftarrow d \rangle \circ I$$
adalah interpretasi yang mencakup domain $D$ dimana:
\begin{enumerate}
  \item Variabel $x$ diberikan nilai elemen $d$ dari $D$
  \item Setiap variabel $y$ (selain $x$) diberi nilai sama dengan nilai berdasar interpretasi $I$
  \item Setiap konstanta, simbol fungsi, dan simbol predikat diberi nilai sesuai dengan $I$
\end{enumerate}

\textbf{Sifat penting}: Jika $I$ adalah interpretasi untuk kalimat berbentuk $(\forall x)\, A$ atau $(\exists x)\, A$, maka $\langle x \leftarrow d \rangle \circ I$ adalah interpretasi yang berlaku untuk $A$.
\end{konsep}

\begin{contoh}
$I$ adalah interpretasi yang meliputi bilangan integer, dengan $x = 1$, $y = 2$. Maka perluasan interpretasi $\langle x \leftarrow 3 \rangle \circ I$ akan memberikan nilai: $x = 3$, $y = 2$.
\end{contoh}

\subsection{Aturan Semantik untuk Kuantifier}

\begin{konsep}
\textbf{Aturan $\forall$ (FOR ALL)}:
\begin{itemize}
  \item $(\forall x)\, A$ bernilai \textbf{TRUE} berdasarkan $I$ jika: \underline{untuk setiap elemen $d$} dari domain $D$, subkalimat $A$ bernilai TRUE berdasarkan $\langle x \leftarrow d \rangle \circ I$
  \item $(\forall x)\, A$ bernilai \textbf{FALSE} berdasarkan $I$ jika: \underline{ada elemen $d$} dari domain $D$ sedemikian sehingga $A$ bernilai FALSE berdasarkan $\langle x \leftarrow d \rangle \circ I$
\end{itemize}
\end{konsep}

\begin{konsep}
\textbf{Aturan $\exists$ (FOR SOME)}:
\begin{itemize}
  \item $(\exists x)\, A$ bernilai \textbf{TRUE} berdasarkan $I$ jika: \underline{ada elemen $d$} dari domain $D$ sedemikian sehingga $A$ bernilai TRUE berdasarkan $\langle x \leftarrow d \rangle \circ I$
  \item $(\exists x)\, A$ bernilai \textbf{FALSE} berdasarkan $I$ jika: \underline{untuk setiap elemen $d$} dari domain $D$, subkalimat $A$ bernilai FALSE berdasarkan $\langle x \leftarrow d \rangle \circ I$
\end{itemize}
\end{konsep}

\begin{contoh}
$A : (\exists x)\, p(x, y)$

Diberikan interpretasi $I$ yang meliputi himpunan bilangan integer positif, dengan $y = 2$ dan $p$: relasi ``kurang dari'' ($p_I(d_1, d_2) = d_1 < d_2$).

Berdasarkan aturan $\exists$: $(\exists x)\, p(x, y)$ bernilai TRUE jika ada elemen $d$ dari $D$ sehingga $p(x, y)$ bernilai TRUE berdasarkan $\langle x \leftarrow d \rangle \circ I$.

Ambil $d = 1$, maka berdasarkan interpretasi $\langle x \leftarrow 1 \rangle \circ I$, diperoleh $p(1, 2)$ yaitu $1 < 2$ adalah \textbf{TRUE}.

Jadi $(\exists x)\, p(x, y)$ bernilai TRUE berdasarkan $I$.
\end{contoh}

\begin{contoh}
$B : \text{IF}\; (\forall x)(\exists y)\, p(x, y) \;\text{THEN}\; p(a, f(a))$

Interpretasi $I$: domain = bilangan real positif, $a = 1$, $f_I(d) = \sqrt{d}$, $p_I(d_1, d_2) = (d_1 \neq d_2)$.

\textbf{Langkah 1: Evaluasi Konsekuen} (karena lebih sederhana)\\
$p(a, f(a)) = p(1, \sqrt{1}) = (1 \neq 1)$ adalah \textbf{FALSE}

\textbf{Langkah 2: Evaluasi Antisenden}\\
$(\forall x)(\exists y)\, p(x, y)$: untuk setiap $d_1$ dari $D$, ada $d_2$ sehingga $d_1 \neq d_2$.

Ambil $d_2 = d_1 + 1$, maka $d_1 \neq d_1 + 1$ adalah TRUE untuk semua $d_1$.

Jadi antisenden bernilai \textbf{TRUE} tetapi konsekuen bernilai \textbf{FALSE}, maka kalimat $B$ bernilai \textbf{FALSE} berdasarkan $I$.
\end{contoh}
