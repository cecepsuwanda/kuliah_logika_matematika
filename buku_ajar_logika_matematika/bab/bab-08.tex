\begin{aktivitas}
  \item Install SWI-Prolog. Tulis program silsilah keluarga (ayah, ibu, kakek, nenek) dengan fakta dan aturan.
  \item Jalankan query untuk menanyakan siapa kakek dari siapa.
  \item Instal SWI-Prolog (Section~8.1). Tulis 5 fakta tentang keluarga Anda menggunakan predikat \texttt{parent/2}, \texttt{male/1}, \texttt{female/1}. Query: siapa anak dari X?
  \item Tulis aturan Prolog untuk \texttt{sibling(X, Y)}, \texttt{grandparent(X, Y)}, dan \texttt{ancestor(X, Y)} (Section~8.2). Uji dengan query menggunakan basis pengetahuan silsilah keluarga (Section~8.4).
  \item Lakukan trace unifikasi (Section~8.3) untuk query \texttt{?- ancestor(tom, jim).} pada basis pengetahuan di studi kasus. Tuliskan setiap langkah substitusi dan backtracking yang terjadi.
  \item Buat basis pengetahuan Prolog sederhana untuk domain selain keluarga (misalnya hewan, kota, atau makanan). Definisikan minimal 5 fakta dan 3 aturan. Query 3 pertanyaan.
\end{aktivitas}

\begin{latihan}
  \item Tulis fakta-fakta Prolog untuk basis pengetahuan berikut: ``Ali suka matematika. Budi suka informatika. Cici suka fisika. Ali suka informatika.'' Tulis query untuk mencari: siapa yang suka informatika?
  \item Tulis aturan Prolog: ``X adalah rekan Y jika keduanya menyukai mata kuliah yang sama.'' Gunakan basis fakta dari soal 1. Jalankan query \texttt{?- rekan(ali, Y).}
  \item Jelaskan proses unifikasi dan backtracking yang terjadi ketika Prolog mengevaluasi query \texttt{?- ancestor(X, jim).} pada basis pengetahuan di Section~8.4. Gambarkan pohon pencarian.
  \item Bandingkan paradigma deklaratif (Prolog) dengan paradigma imperatif (Python/Java). Berikan contoh masalah yang lebih mudah diselesaikan dengan masing-masing paradigma.
  \item Hubungkan konsep fakta dan aturan Prolog dengan logika predikat (Bab V): tunjukkan korespondensi antara klausa Prolog dan formula logika predikat.
  \item \textbf{Refleksi}: Apa kelebihan dan keterbatasan paradigma logic programming dibandingkan pemrograman imperatif yang telah Anda pelajari? Dalam domain apa Prolog paling cocok?
\end{latihan}

\begin{asesmen}
\textbf{Instrumen Evaluasi Kinerja untuk Sub-CPMK 6.1}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Dalam Prolog, \texttt{parent(tom, bob).} adalah:
  \begin{enumerate}
    \item Aturan
    \item Fakta
    \item Query
    \item Variabel
  \end{enumerate}
  \item Prolog menyelesaikan query dengan mekanisme:
  \begin{enumerate}
    \item Iterasi dan loop
    \item Unifikasi dan backtracking
    \item Kompilasi dan linking
    \item Parsing dan tokenizing
  \end{enumerate}
  \item Dalam klausa Prolog \texttt{ancestor(X,Y) :- parent(X,Y).}, bagian \texttt{parent(X,Y)} disebut:
  \begin{enumerate}
    \item Head
    \item Body (subgoal)
    \item Query
    \item Atom
  \end{enumerate}
  \item Variabel dalam Prolog selalu dimulai dengan:
  \begin{enumerate}
    \item Huruf kecil
    \item Huruf besar atau underscore
    \item Angka
    \item Tanda kutip
  \end{enumerate}
\end{enumerate}

\textbf{B. Tugas Praktik (Evaluasi Kinerja)}
\begin{itemize}
  \item Buat basis pengetahuan lengkap silsilah keluarga minimal 3 generasi menggunakan predikat \texttt{parent/2}, \texttt{male/1}, \texttt{female/1} (Section~8.4).
  \item Definisikan aturan: \texttt{sibling/2}, \texttt{grandparent/2}, \texttt{ancestor/2}, \texttt{uncle/2} atau \texttt{aunt/2}.
  \item Jalankan 5 query berbeda dan dokumentasikan hasil serta proses unifikasi/backtracking untuk salah satu query.
  \item Tuliskan korespondensi antara setiap aturan Prolog dengan formula logika predikat.
\end{itemize}

\begin{rangkuman}
Bab ini memperkenalkan Prolog dan \logic{logic programming}. Program Prolog terdiri atas \emph{fakta}, \emph{aturan}, dan \emph{query}; eksekusi didasarkan pada unifikasi dan resolution. SWI-Prolog dapat digunakan untuk praktik menulis program dan menjalankan query dasar.

\textbf{Poin Kunci:}
\begin{itemize}
  \item Prolog adalah bahasa deklaratif: program mendefinisikan relasi (fakta dan aturan), mesin menjawab query.
  \item Fakta adalah predikat yang selalu benar; aturan berbentuk \texttt{Kepala :- Tubuh} (Kepala jika Tubuh).
  \item Query diawali \texttt{?-}; jawaban berupa \texttt{true}/\texttt{false} atau binding variabel.
  \item Unifikasi menemukan substitusi agar dua term sama; resolution dan backtracking adalah mekanisme inferensi Prolog.
  \item Program Prolog bersesuaian dengan klausa Horn dalam logika predikat.
  \item SWI-Prolog dan Learn Prolog Now menyediakan lingkungan praktik dan tutorial.
\end{itemize}

\textbf{Kata Kunci}: \logic{Prolog}, fakta, aturan, query, unifikasi, resolution, backtracking, klausa Horn, SWI-Prolog.
\end{rangkuman}
