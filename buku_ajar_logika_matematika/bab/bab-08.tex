\begin{aktivitas}
  \item \textbf{Basis pengetahuan:} Rancang basis pengetahuan untuk hierarki organisasi kampus (rektor, dekan, ketua prodi, dosen, mahasiswa). Daftarkan minimal 8 fakta ($\Delta$) dan 4 aturan ($\Gamma$) dalam notasi FOL.
  \item \textbf{Resolusi query:} Gunakan KB dari aktivitas sebelumnya. Demonstrasikan forward chaining untuk menjawab query ``Siapa atasan langsung dari mahasiswa X?'' (atau query serupa yang Anda definisikan).
  \item \textbf{Pemodelan aturan:} Modelkan aturan perpustakaan: ``Mahasiswa yang terdaftar dan tidak memiliki denda dapat meminjam buku.'' Tuliskan dalam FOL (fakta dan aturan).
  \item \textbf{Inferensi kuantor:} Pada contoh yang Anda buat, tunjukkan penerapan keempat aturan inferensi kuantor (UI, UG, EI, EG) masing-masing minimal satu kali.
  \item \textbf{CWA vs OWA:} Bandingkan CWA dan OWA pada basis pengetahuan perpustakaan: jika $Terdaftar(MahasiswaY)$ tidak ada dalam KB, apa kesimpulan sistem di bawah CWA dan di bawah OWA?
  \item \textbf{Batasan FOL:} Identifikasi minimal dua batasan FOL yang relevan dalam pemodelan sistem informasi akademik. Jelaskan cara mengatasinya atau paradigma alternatif.
\end{aktivitas}

\begin{latihan}
  \item Translasi ke FOL: Tuliskan fakta berikut dalam bentuk predikat $Suka(x, y)$: ``Ali suka matematika. Budi suka informatika. Cici suka fisika. Ali suka informatika.''
  \item Definisikan aturan $Rekan(x, y)$: ``$x$ adalah rekan $y$ jika keduanya menyukai mata kuliah yang sama dan $x \neq y$.'' Tunjukkan proses inferensi untuk domain $\{\text{Ali}, \text{Budi}, \text{Cici}\}$ hingga diperoleh pasangan rekan.
  \item Diberikan aturan $\forall x, y\, (OrangTua(x, y) \wedge Laki(x) \rightarrow Ayah(x, y))$. Jika KB berisi $OrangTua(Hasan, Ali)$ dan $Laki(Hasan)$, tunjukkan langkah inferensi (UI dan Modus Ponens) untuk membuktikan $Ayah(Hasan, Ali)$.
  \item Modelkan aturan e-commerce dalam FOL: ``Pengguna yang login, terverifikasi, dan memiliki saldo $\ge$ harga item dapat melakukan pembelian.'' Tuliskan fakta dan aturan, lalu demonstrasikan inferensi untuk satu contoh.
  \item Jelaskan perbedaan Universal Instantiation (UI) dan Existential Instantiation (EI). Berikan satu contoh penggunaan masing-masing.
  \item Apa yang terjadi jika dalam backward chaining konklusi yang dicari mengarah ke fakta yang tidak ada dalam KB, dan sistem menggunakan CWA?
  \item \textbf{Refleksi:} (a) Berikan contoh masalah dunia nyata yang lebih mudah direpresentasikan dengan basis pengetahuan (deklaratif) daripada algoritma imperatif. (b) Mengapa basis data relasional (SQL) mengadopsi CWA sedangkan web semantik (OWL) mengadopsi OWA? Apa implikasinya bagi desain sistem?
\end{latihan}

\begin{asesmen}
\textbf{Instrumen Evaluasi Kinerja (Sub-CPMK 2.3)}

\textbf{A. Pilihan Ganda}
\begin{enumerate}
  \item Dalam skema basis pengetahuan, pernyataan kebenaran spesifik tentang entitas (tanpa variabel) disebut:
  \begin{enumerate}
    \item Aturan
    \item Fakta atomik (ground atom)
    \item Variabel terikat
    \item Kuantor eksistensial
  \end{enumerate}
  Jawaban: (b)

  \item Untuk memodelkan aturan umum ``Semua manusia adalah mortal'' dalam FOL, bentuk yang tepat adalah:
  \begin{enumerate}
    \item Konjungsi tanpa variabel
    \item Implikasi berkuantor universal $\forall x\, (P(x) \rightarrow Q(x))$
    \item Biimplikasi
    \item Hukum De Morgan
  \end{enumerate}
  Jawaban: (b)

  \item Universal Instantiation (UI) memungkinkan:
  \begin{enumerate}
    \item Menggeneralisasi fakta spesifik menjadi aturan universal
    \item Menurunkan fakta spesifik $P(c)$ dari $\forall x\, P(x)$
    \item Membuktikan keberadaan entitas dengan properti tertentu
    \item Menggabungkan dua aturan menjadi satu
  \end{enumerate}
  Jawaban: (b)

  \item Di bawah Asumsi Dunia Tertutup (CWA), jika $Laki(Budi)$ tidak ada dalam KB, maka:
  \begin{enumerate}
    \item $Laki(Budi)$ bernilai benar
    \item $Laki(Budi)$ bernilai tidak diketahui
    \item $Laki(Budi)$ dianggap salah
    \item Sistem menghasilkan error
  \end{enumerate}
  Jawaban: (c)

  \item Strategi penalaran yang memulai dari query dan mencari aturan yang konsekuennya cocok dengan query secara mundur disebut:
  \begin{enumerate}
    \item Forward chaining
    \item Backward chaining
    \item Modus Ponens
    \item Universal Generalization
  \end{enumerate}
  Jawaban: (b)
\end{enumerate}

\textbf{B. Tugas Praktik}
\begin{enumerate}
  \item Buat skenario basis pengetahuan untuk organisasi kepanitiaan mahasiswa. Daftarkan $\Delta$ (fakta) dan $\Gamma$ (aturan). Tunjukkan alur inferensi untuk menjawab query ``Siapa yang bertanggung jawab atas tugas X?'' (atau query serupa).
  \item Bandingkan hasil query pada KB yang sama di bawah CWA dan OWA. Berikan satu contoh query yang menghasilkan jawaban berbeda di bawah kedua asumsi.
\end{enumerate}

\textbf{Rubrik}: Merujuk Lampiran Buku.
\end{asesmen}

\begin{rangkuman}
Bab ini membahas penerapan logika predikat orde pertama untuk memodelkan masalah komputasi dalam bentuk basis pengetahuan.

\textbf{Poin Kunci:}
\begin{itemize}
  \item \textbf{Basis Pengetahuan} terdiri dari fakta ($\Delta$, pernyataan atomik tanpa variabel) dan aturan ($\Gamma$, implikasi berkuantor universal). Pemodelan memerlukan identifikasi entitas, properti, dan relasi dari domain.
  \item \textbf{Arity} predikat menentukan jumlah argumennya: arity-1 untuk properti, arity-2 untuk relasi biner, dan arity-$n$ untuk relasi multi-argumen.
  \item \textbf{Empat aturan inferensi kuantor} (UI, UG, EI, EG) memungkinkan penalaran formal dengan membuka atau memasang kuantor.
  \item \textbf{Resolusi query} menggunakan forward chaining (data-driven) atau backward chaining (goal-driven) untuk menurunkan kesimpulan baru dari fakta dan aturan.
  \item \textbf{Asumsi Dunia Tertutup} (CWA) menganggap fakta yang tidak ada dalam KB sebagai salah, sementara \textbf{Asumsi Dunia Terbuka} (OWA) menganggapnya tidak diketahui.
  \item FOL memiliki batasan dalam menangani ketidakpastian, perubahan pengetahuan, dan kompleksitas komputasi, tetapi tetap menjadi fondasi bagi representasi pengetahuan modern.
\end{itemize}

\textbf{Kata Kunci}: basis pengetahuan, fakta, aturan, predikat, arity, instansiasi universal, generalisasi, forward chaining, backward chaining, CWA, OWA
\end{rangkuman}
