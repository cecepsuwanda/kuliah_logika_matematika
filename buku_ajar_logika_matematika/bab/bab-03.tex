\begin{asesmen}
\textbf{Instrumen untuk Sub-CPMK 1.1--1.3}

\textbf{A. Pilihan Ganda}

\begin{enumerate}
  \item Manakah nilai kebenaran $p \rightarrow q$ ketika $p$ salah dan $q$ benar? (a) Benar \quad (b) Salah. \quad Jawaban: (a) Benar.
  \item Manakah yang \emph{bukan} proposisi? (a) ``$2+2=4$.'' \quad (b) ``Apakah hari hujan?'' \quad (c) ``Semua bilangan prima ganjil.'' \quad (d) ``$x < 0$.''
  \item $p \leftrightarrow q$ benar jika dan hanya jika: (a) $p$ dan $q$ sama nilai kebenarannya \quad (b) $p$ benar \quad (c) $q$ salah \quad (d) $p \wedge q$ benar.
  \item Negasi dari $p \wedge q$ menurut Hukum De Morgan adalah: (a) $\neg p \wedge \neg q$ \quad (b) $\neg p \vee \neg q$ \quad (c) $p \vee q$ \quad (d) $\neg(p \vee q)$.
  \item Kontraposisi dari $p \rightarrow q$ adalah: (a) $q \rightarrow p$ \quad (b) $\neg p \rightarrow \neg q$ \quad (c) $\neg q \rightarrow \neg p$ \quad (d) $\neg q \rightarrow p$.
\end{enumerate}

\textbf{B. Essay singkat}

\begin{enumerate}
  \item Jelaskan perbedaan proposisi atomik dan proposisi majemuk. Berikan masing-masing dua contoh.
  \item Buat tabel kebenaran untuk $\neg(p \vee q)$ dan $(\neg p) \wedge (\neg q)$. Buktikan Hukum De Morgan kedua.
  \item Berdasarkan studi kasus sistem lampu (Section~3.5), modifikasi kondisi agar lampu menyala dengan operator XOR. Buat tabel kebenaran dan jelaskan perbedaannya.
\end{enumerate}

\begin{rangkuman}
Bab ini membahas proposisi dan operator logika sebagai fondasi logika matematika.

\textbf{Poin Kunci:}
\begin{itemize}
  \item Proposisi adalah pernyataan deklaratif yang bernilai benar atau salah. Proposisi terbagi menjadi proposisi atomik (tunggal) dan proposisi majemuk (gabungan dengan operator).
  \item Lima operator logika: negasi ($\neg$), konjungsi ($\wedge$), disjungsi ($\vee$), implikasi ($\rightarrow$), bikondisional ($\leftrightarrow$). Prioritas dari tertinggi: $\neg, \wedge, \vee, \rightarrow, \leftrightarrow$.
  \item Tabel kebenaran mendaftar semua $2^n$ kombinasi nilai variabel dan hasil formula; digunakan untuk menentukan tautologi (selalu benar), kontradiksi (selalu salah), dan kontingensi.
  \item Hukum De Morgan: $\neg(p \wedge q) \equiv \neg p \vee \neg q$ dan $\neg(p \vee q) \equiv \neg p \wedge \neg q$ --- penting untuk negasi pernyataan majemuk.
  \item Varian implikasi: konvers ($q \rightarrow p$), invers ($\neg p \rightarrow \neg q$), kontraposisi ($\neg q \rightarrow \neg p$). Hanya kontraposisi yang ekuivalen dengan implikasi asli.
  \item Penerapan dalam pemrograman: operator Boolean (\texttt{and}, \texttt{or}, \texttt{not}), struktur \texttt{if}; studi kasus sistem lampu dan kondisi kelulusan.
\end{itemize}

\textbf{Kata Kunci}: proposisi, operator logika, tabel kebenaran, negasi, konjungsi, disjungsi, implikasi, bikondisional, tautologi, kontradiksi, kontingensi, Hukum De Morgan, konvers, invers, kontraposisi
\end{rangkuman}
