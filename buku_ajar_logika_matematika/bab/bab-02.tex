\begin{asesmen}
\textbf{Instrumen Penilaian untuk Sub-CPMK Bab 2}

\textbf{A. Pilihan Ganda}

\begin{enumerate}
  \item Pernyataan yang bernilai benar atau salah, tetapi tidak keduanya, dalam logika matematika disebut:
  \begin{enumerate}
    \item Premis
    \item Proposisi
    \item Kesimpulan
    \item Aturan inferensi
  \end{enumerate}

  \item Dalam konteks logika, argumen dikatakan valid jika:
  \begin{enumerate}
    \item Premis-premisnya benar
    \item Kesimpulannya mengikuti secara logis dari premis
    \item Jumlah premis lebih dari satu
    \item Kesimpulannya populer di masyarakat
  \end{enumerate}

  \item Tokoh yang memperkenalkan logika predikat dengan kuantifier adalah:
  \begin{enumerate}
    \item Aristoteles
    \item George Boole
    \item Gottlob Frege
    \item Wittgenstein
  \end{enumerate}

  \item Dalam Python, ekspresi \texttt{x < 10 and y > 5} menggunakan konsep logika:
  \begin{enumerate}
    \item Disjungsi
    \item Konjungsi
    \item Implikasi
    \item Negasi
  \end{enumerate}
\end{enumerate}

\textbf{B. Essay}

\begin{enumerate}
  \item Jelaskan peran logika matematika dalam tiga bidang informatika yang berbeda, berikan contoh konkret masing-masing.
  \item Uraikan lintasan belajar dari proposisi (Bab III) hingga Prolog (Bab VIII) dan jelaskan mengapa urutan tersebut penting bagi pemahaman mahasiswa.
\end{enumerate}

\begin{rangkuman}
Bab ini menyajikan fondasi teoretis logika matematika, peranannya dalam informatika, dan peta konsep materi yang akan dipelajari.

\textbf{Poin Kunci:}
\begin{itemize}
  \item Logika matematika mempelajari penalaran formal, argumen valid, serta aturan inferensi; konsep kunci meliputi proposisi, premis, kesimpulan, dan argumen valid (Tabel~\ref{tab:konsep-logika}).
  \item Perkembangan logika dari Aristoteles (silogistik), kaum Stoik (proposisional), Boole (aljabar Boolean), Frege (kuantifier), hingga Wittgenstein (tabel kebenaran) membentuk fondasi logika simbolik dan komputasi (Tabel~\ref{tab:timeline-logika}).
  \item Logika diterapkan dalam lima bidang informatika: pemrograman (Boolean, struktur kontrol), basis data (SQL, predikat), AI (inferensi, sistem pakar), verifikasi perangkat lunak, dan perangkat keras digital (gerbang logika).
  \item Studi kasus: sistem login ($p_\text{user} \wedge p_\text{pass}$), query basis data (predikat konjungsi), dan sistem pakar (modus ponens).
  \item Peta konsep buku berurutan: logika proposisional $\to$ ekuivalensi dan tautologi $\to$ logika predikat $\to$ inferensi dan bukti $\to$ induksi matematika $\to$ logic programming (Prolog).
\end{itemize}

\textbf{Kata Kunci}: \logic{Logika matematika}, Proposisi, Premis, Kesimpulan, Argumen valid, Aturan inferensi, Gerbang logika, Aljabar Boolean, Peta konsep
\end{rangkuman}
